\subsection{Semirings}\label{sec:semirings}

We first recall the definition of semirings and m-semirings, and give examples of such
structures. We also recall the definition of annotated relations and queries,
and how to evaluate queries on annotated relations. Finally, we recall the
definition of aggregate queries and their semantics.

\begin{definition}[Semiring]
  A \emph{semiring} is an algebraic structure denoted by $(\mathbb K,\oplus,\otimes,\mathbb 0,\mathbb 1)$ where $\mathbb K$ is a set, $\oplus$ and $\otimes$ are binary operations on $\mathbb K$, and $\mathbb 0,\mathbb 1\in \mathbb K$, such that:
  \begin{itemize}
    \item $(\mathbb K,\oplus,\mathbb 0)$ is a commutative monoid and $(\mathbb K,\otimes,\mathbb 1)$ is a (not necessarily commutative) monoid;
    \item $\otimes$ distributes over $\oplus$: $\forall~a,b,c\in K$, $a\otimes (b\oplus c) = (a\otimes b)\oplus (a\otimes c)$ and $(a\oplus b)\otimes c = (a\otimes c)\oplus (b\otimes c)$;
    \item $\mathbb 0$ is an annihilator for $\otimes$: $\forall a\in K$, $a\otimes \mathbb 0 = \mathbb 0 \otimes a = \mathbb 0$.
  \end{itemize}
\end{definition}

\begin{example}\label{ex:semirings}
  Some example semirings are:
  \begin{itemize}
    \item The counting semiring $\mathbb N$: $(\mathbb N,+,\times,0,1)$.
    \item The Boolean semiring $\mathbb B$: $(\{\top,\bot\},\lor,\land,\bot,\top)$.
    \item The Why-provenance semiring over a set of variables $X$: $({2^2}^X,\cup,\Cup,\emptyset,\{\emptyset\})$ where $\Cup$ is defined as follows: $A\Cup B := \{a\cup b \mid a\in A, b\in B\}$.
  \end{itemize}
\end{example}

We extend the definition of semiring with a monus operation $\ominus$:
\begin{definition}[m-semiring]
  An \emph{m-semiring}, or \emph{semiring with monus}, denoted by $(\mathbb K,\oplus,\otimes,\ominus,\mathbb 0,\mathbb 1)$, is a semiring equipped with a binary operation $\ominus: \mathbb K\times \mathbb K\to \mathbb K$, such that $\forall a,b,c\in\mathbb K$:
  \begin{itemize}
    \item $a\oplus(b\ominus a) = b\oplus(a\ominus b)$;
    \item $(a\ominus b)\ominus c = a\ominus(b\oplus c)$; and
    \item $a\ominus a = \mathbb 0\ominus a=\mathbb 0$. 
  \end{itemize}
\end{definition}

Additionally to the properties of their definitions, semirings may have other properties. Useful ones are:
\begin{itemize}
  \item \textbf{Commutativity of $\otimes$}: $\forall a,b\in \mathbb K$, $a\otimes b = b\otimes a$.
  \item \textbf{Idempotence}: $\forall a\in \mathbb K$, $a\oplus a = a$.
  \item \textbf{Absorptivity} $\forall a\in \mathbb K$, $a\oplus (a\otimes b) = a\oplus (b\otimes a)=a$.
  \item \textbf{Right distributivity of $\ominus$ over $\oplus$}: $\forall a,b,c\in \mathbb K$, $(a\oplus b)\ominus c = (a\ominus c)\oplus (b\ominus c)$.
\end{itemize}

\paragraph*{Useful results on m-semirings.} We discuss now some additional results on m-semirings that we use later. In particular, we show that in an m-semiring, additive idempotence is equivalent to $\ominus$ being right-distributive over $\oplus$. We also show that homomorphisms between m-semirings preserve these properties. 
\begin{propositionrep}
  \lean{SemiringWithMonus}{idempotent_iff_add_monus}
  Let $S$ be any m-semiring. $S$ is idempotent if and only if $\ominus$
  is right-distributive over~$\oplus$ in $S$.
\end{propositionrep}
\begin{appendixproof}
  \textbf{($\Rightarrow$)}
  \lean{SemiringWithMonus}{add_monus_of_idempotent}

  if $a\leq b$ then $\exists x,\; a\oplus x=b$.

  \\Conversely, if $a\oplus b = b$, choose $x=b$. And trivially, $a\oplus x =a\oplus b=b$
  $\implies a \leq b$.
  \\$\therefore a\leq b \iff a \oplus b =b$. Therefore, $(S,\leq)$ is indeed a partial order and $\oplus$ is monotone under $\leq$.
  \\Now, $\forall u, a \leq u$ and $b \leq u$
  \\ $a \oplus u = u$ and $b\oplus u =u$.
  \\But, $a\leq a \oplus b$ and $b \leq a \oplus b$

\[

  But, trivially $a\le a\oplus a$, so antisymmetry of $(S,\leq)$ gives $a\oplus a=a$.
  Thus, $$\boxed{a \oplus a=a, \; \forall a\in S}$$

\begin{corollaryrep}[Right-distributivity of $\ominus$ over finite $\oplus$-sums]\label{lem:finite-add-monus}
Let \((\mathbb K,\oplus,\otimes,\ominus,\mathbb{0},\mathbb{1})\) be an
m-semiring in which \(\ominus\) is right-distributive over \(\oplus\).

Then for every \(n \ge 1\) and \(\forall a_1,\dots,a_n,c \in K\),
\[
\Bigl(\bigoplus_{i=1}^n a_i\Bigr)\ominus c
=
\bigoplus_{i=1}^n (a_i \ominus c).
\]
\end{corollaryrep}

\begin{appendixproof}
We proceed by induction on \(n \ge 1\).

For \(n=1\), the identity is trivial.
For \(n=2\), the statement is exactly the assumed right-distributivity law:
\[
(a_1 \oplus a_2)\ominus c = (a_1\ominus c)\oplus(a_2\ominus c).
\]

Assume the identity holds for \(n-1\).
Then
\[
\begin{aligned}
\Bigl(\bigoplus_{i=1}^n a_i\Bigr)\ominus c
&= \Bigl(\bigl(\bigoplus_{i=1}^{n-1} a_i\bigr)\oplus a_n\Bigr)\ominus c \\
&= \Bigl(\bigoplus_{i=1}^{n-1} a_i\Bigr)\ominus c \;\oplus\; (a_n\ominus c) \\
&= \bigoplus_{i=1}^{n-1} (a_i\ominus c) \;\oplus\; (a_n\ominus c)
= \bigoplus_{i=1}^n (a_i\ominus c),
\end{aligned}
\]
where the second equality uses right-distributivity and the third uses the
induction hypothesis.
\end{appendixproof}

Finally, we isolate two elementary inequalities on the residual characterization of $\ominus$ and the natural order defined for commutative $m-$semirings.
\begin{lemmarep}\label{lem:x-y<=x}
  Let $(\mathbb K,\oplus,\otimes,\ominus,\mathbb 0,\mathbb 1)$ be a idempotent commutative m-semiring, where $\ominus$ satisfies the residuation law: 
\begin{equation}\label{eq:residuation-monus}
  x\ominus y \le z \quad\Longleftrightarrow\quad x \le y\oplus z,
\end{equation}
where $\le$ is the natural order $a\le b \iff a\oplus b=b$.

Then $\forall x,y\in \mathbb K$, $x\ominus y \le x$.
\end{lemmarep}

\begin{appendixproof}
By \eqref{eq:residuation-monus} with $z=x$, we have
\[
x\ominus y \le x \quad\Longleftrightarrow\quad x \le y\oplus x.
\]
But $x\le y\oplus x$ holds because $x\oplus(y\oplus x)=(x\oplus x)\oplus y=x\oplus y=y\oplus x$
(using associativity, commutativity and idempotence of $\oplus$).
Hence $x\ominus y\le x$.
\end{appendixproof}

\begin{lemmarep}\label{lem:x<=y+(x-y)}
  Let $(\mathbb K,\oplus,\otimes,\ominus,\mathbb 0,\mathbb 1)$ be a idempotent commutative m-semiring, where $\ominus$ satisfies the residuation law in Equation~\eqref{eq:residuation-monus}.

  Then $\forall x,y\in \mathbb K$, $x \le y\oplus (x\ominus y)$.

\end{lemmarep}

\begin{appendixproof}
Apply \eqref{eq:residuation-monus} with $z=x\ominus y$:
since $x\ominus y\le x\ominus y$ holds trivially, \eqref{eq:residuation-monus} yields
\[
x \le y\oplus (x\ominus y),
\]
as required.
\end{appendixproof}
