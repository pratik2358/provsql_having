\section{Semantics}
\label{sec:semantics}

In this section we give a general \emph{possible-world semantics} for
selection predicates involving comparisons of aggregate values, in
arbitrary m-semirings.

In the approach presented in~\cite{amsterdamer2011provenance}, also used
in the ProvSQL implementation \cite{sen2026provsql},
annotation of aggregate query results rely on a \emph{semimodule} whose
elements are tensor products of data values and semiring elements, with
the aggregation operation, required to be a commutative monoid operation,
being lifted as the semimodule sum. This allows compositionality and some
form of compatibility between semiring operations and the aggregate
values. On the other hand, this restricts aggregate operations, with
common non-associative operations (such as \sql{AVG}) or non-commutative ones
(such as \fakesql{STRING\_AGG} or \fakesql{ARRAY\_AGG}). In contrast, for
generality and because our semantics does not depend on the fine
semimodule structure, we extend to any arbitrary aggregate function on
\emph{sequences} of values. This results in replacing the semimodule
structure with a formal non-commutative algebraic structure.

We fix a commutative m-semiring
$(\mathbb{K},\oplus,\otimes,\ominus,\mathbb{0},\mathbb{1})$, as discussed in the preliminaries, along with a data domain $\mathcal{V}$.
For any set~$S$, we write $\Seq(S)$ the set of finite sequences over $S$.
We consider aggregate functions and comparison operators:
\begin{definition}
An \emph{aggregate function} is a function  $f:\Seq(\mathcal{V})\to\mathcal{V}$.
  For any comparison operator $\op\in\{<,\leq,=,\geq,>,\neq\}$ defined
  over $\mathcal{V}$, i.e., $\op: \mathcal{V}^2\rightarrow
  \{\bot,\top\}$, we denote
\(
  \chi_{\op}:\mathcal{V}\times\mathcal{V}\to \mathbb{K}
  \) the mapping that sends $(a,b)$ to $\mathbb{1}$ if $a\op b$, and to
  $\mathbb{0}$ otherwise.
\end{definition}

\cite{sen2026provsql} provides a provenance semantics over annotated
relations for the multiset relational alegbra extended with
\emph{terminal} aggregation,
which itself is based on that of
\cite{green2008provenance,geerts2010database,amsterdamer2011provenance}.
We extend the semantics of the \emph{selection} operator to allow
comparisons of aggregate values (i.e., non-terminal aggregates) using a
possible-world semantics.
\pierre{We need to add an ordering function on aggregation.}
We now introduce \emph{aggregate comparison atoms} in \emph{selection} conditions:

\begin{definition}
Let $\hat I$ be a $\mathbb{K}$-instance. For $k\in \mathbb{N}$, a $k$-ary
extended relational algebra query $q$, distinct grouping indices $(i_j)_{1\leq j \leq m}$ s.t. $1\leq i_j \leq k$ and any group $\vec v=(v_1,\dots,v_m)\in\mathcal{V}^m$, we define
the sequence of occurrences of tuples in $k$ involved as
$
U^{\hat I}_{q,\vec v} \defeq (\, (u_r,\alpha_r)\in \angsem{\hat I}{q} | (u_{r,i_1},\dots,u_{r,i_m})=\vec v \,)_{1\leq r\leq|\angsem{\hat I}{q}|}$, then for any \emph{possible-world} of $\vec v$, $W\subseteq [|U^{\hat I}_{q,\vec v}|]$, we define its $\mathbb{K}-$annotation as
$
\ann^{\hat I}_{q,\vec v}(W) \defeq
\Bigl( \bigotimes_{U^{\hat I}_{q,\vec v}[r], r\in W}\alpha_r \Bigr)
 \otimes
 \Bigl( \mathbb{1} \ominus \bigoplus_{U^{\hat I}_{q,\vec v}[r], r\in [|U^{\hat I}_{q,\vec v}|]\setminus W}\alpha_r \Bigr)
$
where, empty $\otimes$-products equal $\mathbb{1}$ and empty $\oplus$-sums equal $\mathbb{0}$.

\end{definition}

We now formalise a general semantics for $selection$, assuming that atomic comparisons inside \emph{selection} predicates- $t\op t'$ and $(t:f) \op (t':f')$, where $t, t'$ are terms of max-index $\leq k$ and $f,f'$ are arbitrary aggregate functions, are well-defined: whenever such a comparison is evaluated on $u\in\mathcal V^k$, $t(u)$ and $t'(u)$  must lie in the domain on
which $\op$ is defined (likewise, for $(t:f)\op(t':f')$).
\begin{definition}[Semantics of Selection Predicate]
Fix a $\mathbb K$-instance $\hat I$  and let $q\in \RA_k$.
A \emph{selection predicate} $\psi$ over arity $k$ induces a function
  $\predsem{\hat I}{q}{\psi} : \mathcal V^k \to \mathbb{K}$,
defined by recursion on~$\psi$ as follows:

\smallskip\noindent
For an atomic comparison $\psi$ of the form $t\op t'$:
\[
\predsem{\hat I}{q}{\psi}(u)\defeq
\begin{cases}
  \mathbb 1 & \text{if }(t(u),t'(u))\in R_{\op}\\
\mathbb 0 & \text{otherwise.}
\end{cases}
\]
\smallskip\noindent
Otherwise, $q$ is of the form $\gamma_{i_1,\dots,i_m}[t_1:f_1,\dots,t_n:f_n](q')\in \RA_{k=m+n}$
with $q'\in\RA_{k'}$, distinct grouping indices $(i_j)_{1\le j\le m}$ such that $1\le i_j\le k'$,
and $\psi=(t:f)\ \op\ (t':f')$ or $\psi=(t:f)\ \op\ t'$ where max-index of $t,t'\leq k'$.

\smallskip\noindent

\smallskip\noindent
For $\psi = (t:f) \op (t':f')$,
\[
\predsem{\hat I}{q}{\psi}(u)\defeq
  \bigoplus_{\substack{W\msubseteq U^{\hat I}_{q',\vec v}\\ W\neq\emptyset}}
\Bigl(
\ann^{\hat I}_{q',\vec v}(W)\otimes
\chi_{\op}\bigl(f(\mset{\, t(u) \mid (u,\alpha)\in W\,}
),\ f'(\mset{\, t'(u) \mid (u,\alpha)\in W\,}
)\bigr)
\Bigr)
\]

\smallskip\noindent
and for $\psi = (t:f) \op t'$,
\[
\predsem{\hat I}{q}{\psi}(u)\defeq
  \bigoplus_{\substack{W\msubseteq U^{\hat I}_{q',\vec v}\\ W\neq\emptyset}}
\Bigl(
\ann^{\hat I}_{q',\vec v}(W)\otimes
\chi_{\op}\bigl(f(\mset{\, t(u) \mid (u,\alpha)\in W\,}
),\ t'(u)\bigr)
\Bigr)
\]
where, for $u=(\vec v,a_1,\dots,a_n)\in\mathcal V^{m+n}$ ,
we write $\vec v=(v_1,\dots,v_m)$ for the group key.


\smallskip\noindent
Now, for $u\in\mathcal V^k$, let $\psi_1$ and $\psi_2$ be arbitrary selection predicates. Then,
\[
\predsem{\hat I}{q}{\psi_1\land \psi_2}(u)\defeq
\predsem{\hat I}{q}{\psi_1}(u)\otimes \predsem{\hat I}{q}{\psi_2}(u),
\]
\[
\predsem{\hat I}{q}{\psi_1\lor \psi_2}(u)\defeq
\predsem{\hat I}{q}{\psi_1}(u)\oplus \predsem{\hat I}{q}{\psi_2}(u),
\]
\[
\predsem{\hat I}{q}{\neg \psi}(u)\defeq \mathbb 1 \ominus \predsem{\hat I}{q}{\psi}(u).
\]

\end{definition}


\begin{definition}[Provenance for HAVING-like queries]Let $\hat I$ be a $\mathbb K$-instance, let $k\in\NN$, let $q\in\RA_k$, and let $\psi$ be
a well-formed selection predicate over arity $k$.
We define:
% \[
% \angsem{\hat I}{\sigma_\psi(q)} \defeq
% \mset{\, (u,\alpha\otimes \predsem{\hat I}{q}{\psi}(u)) \mid (u,\alpha)\in \angsem{\hat I}{q}\,}.
% \]
%
\[
\angsem{\hat I}{\sigma_\psi(q)} \defeq
\begin{cases}
\mset{\, (u,\alpha\otimes \predsem{\hat I}{q}{\psi}(u)) \mid (u,\alpha)\in \angsem{\hat I}{q}\,}
& \text{if $\psi$ contains no aggregate-comparison atom,}\\[1ex]
\mset{\, (u,\predsem{\hat I}{q}{\psi}(u)) \mid (u,\alpha)\in \angsem{\hat I}{q}\,}
& \text{otherwise.}
\end{cases}
\]
\end{definition}\label{def:semantics}

\smallskip\noindent\emph{Note:}
For any aggregate comparison atom, if $U^{\hat I}_{q',\vec v}=\emptyset$ then every output
tuple $u$ with group key $\vec v$, we have $\predsem{\hat I}{q}{\theta}(u)=\mathbb 0$.
Since the possible-world semantics
ranges over non-empty worlds only.
So multiplying the possible world provenance
expression with $\alpha$( which is a $\delta(\cdot)$-type provenance expression to enforce group existence) is redundant.
