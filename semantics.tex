\section{Semantics}
\label{sec:semantics}
\pierre{1 page}

In the semimodule approach presented in~\cite{amsterdamer2011provenance}, provenance for aggregation is made algebraically uniform and compositional by lifting the aggregate to act on $\mathbb{K}$-annotated input tuples. 
This lifting is canonical only for aggregates with the right algebraic structure (specifically monoid aggregates, or aggregates reducible to monoid states such as \sql{AVG} via \sql{SUM} and \sql{COUNT}).

% In this section we give a general \emph{possible world semantics} for selection predicates involving comparisons between aggregates.
% This semantics does not need an algebraic lifting of the aggregate function.
% Instead, for each possible subset of tuples, called a \emph{possible world}, in an aggregate group we evaluate the aggregate as an arbitrary function over the sequence of values and then restrict provenance to the satisfying cases only.
% Thus any SQL aggregate function (average, median/percentile, array/JSON aggregates, custom UDFs, etc.) can be handled in this semantics.
% Our semantics naturally support arbitrary arithmetic over arbitrary aggregate functions in \sql{HAVING}-like conditions; they are evaluated on the aggregate values in each possible world, and then the provenance is accumulated only for the satisfying  possible-worlds.
In this section we give a general \emph{possible world semantics} for selection predicates involving comparisons between arbitrary aggregates or even comparisons between arbitrary arithmetic over the aggregate function(s).
We fix $(\mathbb{K},\oplus,\otimes,\ominus,\delta,\mathbb{0},\mathbb{1})$ as our
annotation structure, as discussed in the preliminaries, along with a finite data domain $\mathcal{V}$.
For any set $S$, we write $\Seq(S)$ for the set of finite sequences over $S$.
We define arbitrary aggregate functions as:
\begin{definition}[Arbitrary Aggregate Function]
  An arbitrary aggregate function is a function  $f:\Seq(\mathcal{V})\to \mathcal{A}$, for some output domain $\mathcal{A}$.
\end{definition}
Now, monoid aggregate functions (\sql{SUM}, \sql{COUNT}, etc.) can be treated as arbitrary aggregate functions in our semantics.
For comparing the result of two aggregates we define a comparison operator $\op$:
\begin{definition}[Comparison Operator]
  For any $\op\in\{=,\neq,<,\leq,>,\geq\}$ defined for a domain $\mathcal{A}$, we define
\[
  \chi_{\op}:\mathcal{A}\times\mathcal{A}\to \mathbb{K},
\qquad
  \chi_{\op}(a,b)\defeq
\begin{cases}
  \mathbb{1}\in \mathbb{K} & \text{if } a \op b,\\
  \mathbb{0} \in \mathbb{K} & \text{otherwise},
\end{cases}
\]
where $\mathbb{K}$ is any commutative $m-$semiring.
\end{definition}
We extend the semiring-provenance semantics of \emph{selection} in \cite{sen2026provsql} with possible-world semantics when \emph{selection}-predicates involve boolean combinations of comparisons between results of arbitrary aggregate functions.

\begin{definition}[Possible-World Semantics]
Let $\hat I$ be a $\mathbb{K}$-instance. For $k\in \mathbb{N}$, a subquery $q\in\RA_k$, distinct grouping indices $(i_j)_{1\leq j \leq m}$ s.t. $1\leq i_j \leq k$ and any group key $\vec v=(v_1,\dots,v_m)\in\mathcal{V}^m$, if we define
the sequence of occurences of tuples involved as 
$
((U^{\hat I}_{q,\vec v})_r)_{1\leq r\leq|\angsem{\hat I}{q}|} \defeq (\, (u_r,\alpha_r)\in \angsem{\hat I}{q} | (u_{r,i_1},\dots,u_{r,i_m})=\vec v \,)_{1\leq r\leq|\angsem{\hat I}{q}|}$, then for any \emph{possible-world} of $\vec v$, $W\subseteq [|((U^{\hat I}_{q,\vec v})_r)|]$, we define its $\mathbb{K}-$annotation as 
$
\ann^{\hat I}_{q,\vec v}(W) \defeq
\Bigl( \bigotimes_{r\in W}(U^{\hat I}_{\substack{q,\vec v}})_r[\alpha] \Bigr)
 \otimes
 \Bigl( \mathbb{1} \ominus \bigoplus_{r\in [|((U^{\hat I}_{q,\vec v})_r)|]\setminus W}(U^{\hat I}_{q,\vec v})_r[\alpha] \Bigr)
$
where, empty $\otimes$-products equal $\mathbb{1}$ and empty $\oplus$-sums equal $\mathbb{0}$.

\end{definition}

We now formalise a general semantics for $selection$, assuming that atomic comparisons inside \emph{selection} predicates- $t\op t'$ and $(t:f) \op (t':f')$, where $t, t'$ are terms of max-index $\leq k$ and $f,f'$ are arbitrary aggregate functions, are well-defined: whenever such a comparison is evaluated on $u\in\mathcal V^k$, $t(u)$ and $t'(u)$  must lie in the domain on
which $\op$ is defined (likewise, for $(t:f)\op(t':f')$).
\begin{definition}[Semantics of Selection Predicate]
Fix a $\mathbb K$-instance $\hat I$  and let $q\in \RA_k$.
A \emph{selection predicate} $\psi$ over arity $k$ induces a function
  $\predsem{\hat I}{q}{\psi} : \mathcal V^k \to \mathbb{K}$,
defined by recursion on~$\psi$ as follows:

\smallskip\noindent
For an atomic comparison $\psi$ of the form $t\op t'$:
\[
\predsem{\hat I}{q}{\psi}(u)\defeq
\begin{cases}
  \mathbb 1 & \text{if }(t(u),t'(u))\in R_{\op}\\
\mathbb 0 & \text{otherwise.}
\end{cases}
\]
\smallskip\noindent
Otherwise, $q$ is of the form $\gamma_{i_1,\dots,i_m}[t_1:f_1,\dots,t_n:f_n](q')\in \RA_{k=m+n}$
with $q'\in\RA_{k'}$, distinct grouping indices $(i_j)_{1\le j\le m}$ such that $1\le i_j\le k'$, 
and $\psi=(t:f)\ \op\ (t':f')$ or $\psi=(t:f)\ \op\ t'$ where max-index of $t,t'\leq k'$.

\smallskip\noindent

\smallskip\noindent
For $\psi = (t:f) \op (t':f')$,
\[
\predsem{\hat I}{q}{\psi}(u)\defeq
\bigoplus_{\substack{W\subseteq [|((U^{\hat I}_{q',\vec v})_r)|]\\ W\neq\emptyset}}
\Bigl(
\ann^{\hat I}_{q',\vec v}(W)\otimes
\chi_{\op}\bigl(f(t(((U^{\hat I}_{q',\vec v})_r[u])_{r\in W})),\ f'(t'(((U^{\hat I}_{q',\vec v})_r[u])_{r\in W})) \bigr)
\Bigr)
\]

\smallskip\noindent
and for $\psi = (t:f) \op t'$,
\[
\predsem{\hat I}{q}{\psi}(u)\defeq
\bigoplus_{\substack{W\subseteq [|((U^{\hat I}_{q',\vec v})_r)|]\\ W\neq\emptyset}}
\Bigl(
\ann^{\hat I}_{q',\vec v}(W)\otimes
\chi_{\op}\bigl(f(t(((U^{\hat I}_{q',\vec v})_r[u])_{r\in W})),\ t'(u)\bigr)
\Bigr)
\]
where, for $u=(\vec v,a_1,\dots,a_n)\in\mathcal V^{m+n}$ ,
we write $\vec v=(v_1,\dots,v_m)$ for the group key.


\smallskip\noindent
Now, for $u\in\mathcal V^k$, let $\psi_1$ and $\psi_2$ be arbitrary selection predicates. Then,
\[
\predsem{\hat I}{q}{\psi_1\land \psi_2}(u)\defeq
\predsem{\hat I}{q}{\psi_1}(u)\otimes \predsem{\hat I}{q}{\psi_2}(u),
\]
\[
\predsem{\hat I}{q}{\psi_1\lor \psi_2}(u)\defeq
\predsem{\hat I}{q}{\psi_1}(u)\oplus \predsem{\hat I}{q}{\psi_2}(u),
\]
\[
\predsem{\hat I}{q}{\neg \psi}(u)\defeq \mathbb 1 \ominus \predsem{\hat I}{q}{\psi}(u).
\]

\end{definition}


\begin{definition}[Provenance for HAVING-like queries]Let $\hat I$ be a $\mathbb K$-instance, let $k\in\NN$, let $q\in\RA_k$, and let $\psi$ be
a well-formed selection predicate over arity $k$.
We define:
% \[
% \angsem{\hat I}{\sigma_\psi(q)} \defeq
% \mset{\, (u,\alpha\otimes \predsem{\hat I}{q}{\psi}(u)) \mid (u,\alpha)\in \angsem{\hat I}{q}\,}.
% \]
%
\[
\angsem{\hat I}{\sigma_\psi(q)} \defeq
\begin{cases}
\mset{\, (u,\alpha\otimes \predsem{\hat I}{q}{\psi}(u)) \mid (u,\alpha)\in \angsem{\hat I}{q}\,}
& \text{if $\psi$ contains no aggregate-comparison atom,}\\[1ex]
\mset{\, (u,\predsem{\hat I}{q}{\psi}(u)) \mid (u,\alpha)\in \angsem{\hat I}{q}\,}
& \text{otherwise.}
\end{cases}
\]
\end{definition}\label{def:semantics}

\smallskip\noindent\emph{Note:}
For any aggregate comparison atom, if $((U^{\hat I}_{q',\vec v})_r)=()$ then every output
tuple $u$ with group key $\vec v$, we have $\predsem{\hat I}{q}{\psi}(u)=\mathbb 0$.
Since the possible-world semantics
ranges over non-empty worlds only.
So multiplying the possible world provenance
expression with $\alpha$( which is a $\delta(\cdot)$-type provenance expression to enforce group existence) is redundant.
