\section{Main Results}
\label{sec:results}
\pierre{3.5 pages: main theorem, proof arguments, complexity}
We show that our semantics as formalised in Section~\ref{sec:semantics} is compatible with the \sql{JOIN}-based semantics of \sql{HAVING}-like queries when our semiring of choice is absorptive and satisfies distributivity of $\otimes$ over $\ominus$.
\begin{theoremrep}\label{th:correctness}
  For any arity~$2$ query $q$, for any $C\in\mathbb{N}_{\ge1}$, for $\op\in \{\leq,
  =,\geq\}$ consider the queries:
\begin{align*}
    Q^{\op C}_1&=\Pi_{\#1}(\sigma_{\#2\op C}(\gamma_{\#1}[1:+](q)))\\
    Q^{\geq C}_2&=\begin{cases}\Pi_{\#1}(q) & \text{if } C=1,\\[1mm]\Pi_{\#1}(q\bowtie_{\#1=\#3\land\#2<^*\#4}
    q\bowtie_{\#1=\#5\land\#4<^*\#6}\dots\bowtie_{\#1=\#(2C-1)\land\#(2C-2)<^*\#(2C)}q) & \text{if }C\ge 2\end{cases}\\
    Q^{\leq C}_2&= Q^{\geq 1}_2-Q_2^{\geq C+1}\\
Q^{=C}_2&=Q_2^{\geq C} -Q_2^{\geq C+1}
\end{align*}
where, the join-chain inequality $<^*$ is $<$ refined with some tie-breaking condition to comply with multiset-semantics.
    Then:
    \begin{enumerate}[(i)]
      \item
        For any commutative m-semiring $\mathbb{K}$ which is
        absorptive and such that $\otimes$ is
        distributive over $\ominus$, for any
        $\mathbb{K}$-instance $\hat I$,
        $\angsem{\hat I}{Q^{\op C}_1}=\angsem{\hat I}{Q^{\op C}_2}$.
      \item There exists a commutative m-semiring $\mathbb{K}_1$ which is
        idempotent but does
        not have distributivity of $\otimes$ over $\ominus$, and a
        $\mathbb{K}_1$-instance $\hat I_1$ such that
        $\angsem{\hat I_1}{Q^{\op C}_1}\neq\angsem{\hat I_1}{Q^{\op C}_2}$.
      \item There exists a commutative m-semiring $\mathbb{K}_2$ that has
        distributivity of $\otimes$ over $\ominus$ and that is idempotent but not
        absorptive, and a
        $\mathbb{K}_2$-instance $\hat I_2$ such that
        $\angsem{\hat I_2}{Q^{\op C}_1}\neq\angsem{\hat I_2}{Q^{\op C}_2}$.
    \end{enumerate}
\end{theoremrep}
\begin{appendixproof}
We proceed with our proof for part $(i)$ of Theorem~\ref{th:correctness}.
\paragraph*{Proof of (i)}
For any $\mathbb{K}$-instance $\hat I$, we prove the equality  $\angsem{\hat I}{Q^{\op C}_1}=\angsem{\hat I}{Q^{\op C}_2}$ by showing
  that for any arbitrary tuple in the query result the corresponding provenance expressions instantiated on any commutative m-semiring $\mathbb{K}$ under the assumptions of $(i)$, are same.
    $\angsem{\hat I}{Q^{\op C}_1}\text{ and }\angsem{\hat I}{Q^{\op C}_2}$ apply the same relational algebra operators to the same subquery $q$
    and the provenance semantics defined for those operators are identical. The two queries differ only in their implementation of the aggregate comparison (\sql{HAVING}) condition; all other operators are interpreted identically. Hence, it
    suffices to show that for each group produced by
    the grouping operator, the provenance annotation produced because of the aggregate comparison coincides with
    that produced by the self-join query.
    Since \(q\) is an arity~$2$ query, every tuple in $\angsem{\hat I}{q}$ has the form $(u,\alpha)$ where u is $(u_1,u_2)$ and $\alpha\in \mathbb K$.
Fix an arbitrary group-key $\vec v=(\#1(u)=a)$.
% \smallskip\noindent\emph{Note: }\textit{For simplicity, we assume set semantics, i.e, $q$ produces no duplicate output tuples. In this setting, for a fixed group key $a$ selecting $C$ tuples whose second attributes form a strict join chain is the same as requiring the existence of $C-$ distinct tuples in the group. The same proof idea can be extended to multiset-semantics by treating duplicate tuples as distinct tuple occurences (e.g. by indexing inside groups) and refining the strict-ordering predicate in the join chain with this new tie-breaking attribute so that it is possible to select distinct tuple occurences with equal second attributes in increasing order.}

So,
\[U^{\hat I}_{q,(a)}\defeq \{(u,\alpha,o)|(u,\alpha) \text{ is the }o\text{th occurence in }   \angsem{\hat I}{q}\text{ and }(u_{i_1})=(a) \}\]
\paragraph{Provenance of the \sql{HAVING} query using possible-world semantics:}
For the query
$
Q^{\op C}_1
=
\Pi_{\#1}\bigl(
\sigma_{\#2 \op C}(
  \gamma_{\#1}[1:+](q)
)\bigr)
$
\begin{multline}\label{eq:Q1-prov}
\angsem{\hat I}{Q^{\op C}_1}(a)
=\predsem{\hat I}{\gamma_{\#1}[1:+](q)}{\#2\op C}(u)
=\bigoplus_{\substack{W\subseteq [|U^{\hat I}_{q,(a)}|]\\ W\neq\emptyset}}
\Bigl(
\ann^{\hat I}_{q,(a)}(W)\otimes
\chi_{\op}\bigl(f_{\Sigma}(\pi_{1}(((U^{\hat I}_{q,(a)})_r[u])_{r\in W}^\omega)),\ C\bigr)
\Bigr)\\
=\bigoplus_{\substack{W\subseteq [|U^{\hat I}_{q,(a)}|]\\ W\neq\emptyset}}
\Bigl(
\ann^{\hat I}_{q,(a)}(W)\otimes
\chi_{\op}\bigl(\Sigma_{r\in W} 1,\ C\bigr)
\Bigr)
\end{multline}
where for any arbitrary ordering $\omega$ our commutative aggregate \sql{COUNT} is denoted as $f_\Sigma$.
\\
For each possible-world $W\subseteq [|U^{\hat I}_{q,(a)}|]$, if we denote
\begin{align*}
&A_W=\bigotimes_{w\in W}(U^{\hat I}_{q,(a)})_w[\alpha]=\bigotimes_{w\in W}\alpha_w\\
&B_W=\bigoplus_{w\in [|U^{\hat I}_{q,(a)}|]\setminus W}(U^{\hat I}_{q,(a)})_w[\alpha] =\bigoplus_{w\in [|U^{\hat I}_{q,(a)}|]\setminus W}\alpha_w\\
\end{align*}
then,
\begin{align*}
T_{[|U^{\hat I}_{q,(a)}|]}(W)&:=\ A_W \otimes \bigl(\mathbb{1}_{\mathbb{K}} \ominus B_W\bigr)\\
&= A_W\ominus \bigoplus_{w\in [|U^{\hat I}_{q,(a)}|]\setminus W}A_{W\cup\{w\}} \;\textit{[ by distributivity of $\otimes$ over $\ominus$ and then of $\otimes$ over $\oplus$]}
\end{align*}
such that, $\ann^{\hat I}_{q,(a)}(W)=T_{[|U^{\hat I}_{q,(a)}|]}(W).$
% By distributivity of $\otimes$ over $\ominus$ and then of $\otimes$ over $\oplus$,
% \begin{align*}
% T_U(W)
%   &= (A_W\otimes \mathbb{1}_{\mathbb{K}})\ominus (A_W\otimes B_W) \\
%   &= A_W\ominus\Bigl(A_W\otimes \bigoplus_{w\in U\setminus W}\alpha_w \Bigr) \\
% &= A_W\ominus \bigoplus_{w\in U\setminus W}(A_W\otimes \alpha_w ) \\
% &= A_W\ominus \bigoplus_{w\in U\setminus W}A_{W\cup\{w\}}.
% \end{align*}
\lean{Having}{Having.T_eq_mul_one_monus_sum}
Since $\chi_{\op}(|W|\op C)=\mathbb{1}_\mathbb{K}$ iff $|W|\op C$ is true and $\mathbb{0}_\mathbb{K}$ otherwise, \eqref{eq:Q1-prov} becomes
\begin{equation}\label{eq:pw_tw}
\angsem{\hat I}{Q^{\op C}_1}(a)
=
\bigoplus_{\substack{W\subseteq [|U^{\hat I}_{q,(a)}|],\\ W\neq\emptyset,\\|W|\op C}} T_{[|U^{\hat I}_{q,(a)}|]}(W).
\end{equation}
\paragraph{Provenance of the \sql{JOIN} queries:}
For every $W\subseteq [|U^{\hat I}_{q,(a)}|] $ and $C\ge 1$ we define,
\[
S_C([|U^{\hat I}_{q,(a)}|]):=\bigoplus_{\substack{W\subseteq [|U^{\hat I}_{q,(a)}|] \\|W|=C}} A_W
\quad
(\text{with }S_C([|U^{\hat I}_{q,(a)}|])=\mathbb{0}_\mathbb{K}\text{ if }C>|U|).
\]
Consider the query $Q^{\ge C}_2$.  A join-valuation contributing to the output
tuple in $\angsem{\hat I}{Q^{\ge C}_2}$  with key $(a)$ corresponds to choosing $C$ tuples from $\angsem{\hat I}{q}$ with first
attribute $a$ and strictly increasing second attributes (due to the chain of
join conditions $\#2<^*\#4<^*\cdots$) with a tie-breaking occurence attribute as in $U^{\hat I}_{q,(a)}$ so that it is possible to select distinct tuple occurences with equal second attributes in an increasing order. Thus, complying with multiset semantics, this corresponds to choosing a
$C$-element subset $W\subseteq[|U^{\hat I}_{q,(a)}|]$, and
$\Pi_{\#1}$ then takes the $\oplus$-sum over all these subsets.
Therefore,
\[
\angsem{\hat I}{Q^{\ge C}_2}(a)
=S_C([|U^{\hat I}_{q,(a)}|]).
\tag{$J_{\ge}$}
\]
By definition,
\[
Q^{=C}_2 = Q^{\ge C}_2 \ominus Q^{\ge (C+1)}_2,
\qquad
Q^{\le C}_2 = \Pi_{\#1}(q) \ominus Q^{\ge (C+1)}_2,
\]
Hence, using $(J_{\ge})$,
\[
\angsem{\hat I}{Q^{=C}_2}(a)
=
S_C([|U^{\hat I}_{q,(a)}|]) \ominus S_{C+1}([|U^{\hat I}_{q,(a)}|]),
\quad
\angsem{\hat I}{Q^{\le C}_2}(a)
=
S_1([|U^{\hat I}_{q,(a)}|]) \ominus S_{C+1}([|U^{\hat I}_{q,(a)}|]).
\tag{$J_{=,\le}$}
\]
\paragraph{\textbf{When $\op$ is $\ge$ :}}
From \eqref{eq:pw_tw} with $\op = \ge$, we set
$
F_C([|U^{\hat I}_{q,(a)}|])\defeq \bigoplus_{\substack{W\subseteq [|U^{\hat I}_{q,(a)}|]\\|W|\ge C}} T_{[|U^{\hat I}_{q,(a)}|]}(W).
$
So,
\begin{equation}\label{eq:PW_ge_equals_F}
\angsem{\hat I}{Q^{\ge C}_1}(a)=F_C([|U^{\hat I}_{q,(a)}|]).
\tag{$PW_{\ge}$}
\end{equation}
We already showed for the join query that
\[
\angsem{\hat I}{Q^{\ge C}_2}(a)=S_C([|U^{\hat I}_{q,(a)}|])
\tag{$J_{\ge}$}
\]
If we set $U=[|U^{\hat I}_{q,(a)}|]$ and and $(\alpha_u)_{u\in U}=((U^{\hat I}_{q,(a)})_u[\alpha])_{u\in U}^\omega$ then, by Lemma~\ref{lem:FC_recurrence}, for an arbitrary choice of $u\in U$ with $U'=U\setminus\{u\}$, for all $C\ge 1$,
\[
F_C(U)=F_C(U')\ \oplus\ \bigl(F_{C-1}(U')\otimes\alpha_u\bigr).
\tag{R$_F$}
\]
Similarly, by Lemma~\ref{lem:SC_recurrence}, $S_C(U)$ can be shown to satisfy the same recurrence:
\[
S_C(U)=S_C(U')\ \oplus\ \bigl(S_{C-1}(U')\otimes\alpha_u\bigr)
\qquad(C\ge 1).
\tag{R$_S$}
\]
We prove that $F_C(U)=S_C(U)$ by induction on $|U|$,  for all finite $U$ and all $C\ge 1$ \lean{Having}{Having.F_eq_S}.
If $|U|=0$ then $U=\varnothing$ and $F_C(\varnothing)=\mathbb 0=S_C(\varnothing)$ for $C\ge 1$
(empty $\oplus$-sum).
Assume $|U|>0$ and pick $u\in U$, let $U'=U\setminus\{u\}$.
By the induction hypothesis, $F_C(U')=S_C(U')$ and $F_{C-1}(U')=S_{C-1}(U')$.
Applying (R$_F$) and (R$_S$) gives
$
F_C(U)
=F_C(U')\oplus(F_{C-1}(U')\otimes\alpha_u)
=S_C(U')\oplus(S_{C-1}(U')\otimes\alpha_u)
=S_C(U)
$.
Thus $F_C([|U^{\hat I}_{q,(a)}|])=S_C([|U^{\hat I}_{q,(a)}|])$ for all $C\ge 1$. Hence by \eqref{eq:PW_ge_equals_F} and $(J_{\ge})$,
\[
\angsem{\hat I}{Q^{\ge C}_1}(a)=\angsem{\hat I}{Q^{\ge C}_2}(a).
\]
\paragraph{\textbf{When $\op$ is $=$ :}}
For $n\in\mathbb N$ we have,
$
\chi_{=}(n,C)=\chi_{\ge}(n,C)\ \ominus\ \chi_{\ge}(n,C+1)
$. Then for each $W\subseteq [|U^{\hat I}_{q,(a)}|]$,
\begin{align*}
T_{[|U^{\hat I}_{q,(a)}|]}(W)\otimes \chi_{=}( |W|,C)
&=
T_{[|U^{\hat I}_{q,(a)}|]}(W)\otimes\bigl(\chi_{\ge}(|W|,C)\ominus\chi_{\ge}(|W|,C+1)\bigr)\\
&=
(T_{[|U^{\hat I}_{q,(a)}|]}(W)\otimes\chi_{\ge}(|W|,C))\ominus(T_{[|U^{\hat I}_{q,(a)}|]}(W)\otimes\chi_{\ge}(|W|,C+1)),
\end{align*}
by distributivity of $\otimes$ over $\ominus$.
Summing over $W$ and using right-distributivity of $\ominus$ over $\oplus$ (which holds because absorptivity in $\mathbb K$ implies idempotence), we obtain
\begin{align*}
\angsem{\hat I}{Q^{=C}_1}(a)
&=
\bigoplus_{W\subseteq [|U^{\hat I}_{q,(a)}|]}\bigl(T_{[|U^{\hat I}_{q,(a)}|]}(W)\otimes\chi_{=}( |W|,C)\bigr)\\
&=
\left(\bigoplus_{W\subseteq [|U^{\hat I}_{q,(a)}|]} T_{[|U^{\hat I}_{q,(a)}|]}(W)\otimes\chi_{\ge}(|W|,C)\right)
\ominus
\left(\bigoplus_{W\subseteq [|U^{\hat I}_{q,(a)}|]} T_{[|U^{\hat I}_{q,(a)}|]}(W)\otimes\chi_{\ge}(|W|,C+1)\right)\\
&=
\left(\bigoplus_{\substack{W\subseteq [|U^{\hat I}_{q,(a)}|],\\|W|\ge C}} T_{[|U^{\hat I}_{q,(a)}|]}(W)\right)
\ominus
\left(\bigoplus_{\substack{W\subseteq [|U^{\hat I}_{q,(a)}|],\\|W|\ge C+1}} T_{[|U^{\hat I}_{q,(a)}|]}(W)\right)\\
&=
\angsem{\hat I}{Q^{\ge C}_1}(a)\ \ominus\ \angsem{\hat I}{Q^{\ge (C+1)}_1}(a).
\end{align*}
By the already established $\ge$-case,
$\angsem{\hat I}{Q^{\ge C}_1}(a)=S_C([|U^{\hat I}_{q,(a)}|])$ and $\angsem{\hat I}{Q^{\ge (C+1)}_1}(a)=S_{C+1}([|U^{\hat I}_{q,(a)}|])$. Therefore,
$
\angsem{\hat I}{Q^{=C}_1}(a)=S_C([|U^{\hat I}_{q,(a)}|])\ominus S_{C+1}([|U^{\hat I}_{q,(a)}|])=\angsem{\hat I}{Q^{=C}_2}(a)
$ by $(J_{=,\le})$.
\paragraph{\textbf{When $\op$ is $\le$:}}
For $n\ge 1$ we have $\chi_{\le}(n,C)=\chi_{\ge}(n,1)\ \ominus\ \chi_{\ge}(n,C+1)$. Indeed, for $n\ge 1$ the condition $n\le C$ is equivalent to $n\ge 1$ and not $n\ge C+1$. Since groups produced by $\gamma_{\#1}$ are nonempty, only $n=|W|\ge 1$ occur.
Repeating the same argument as in the case where $\op$ is $=$, we obtain
\[
\angsem{\hat I}{Q^{\le C}_1}(a)
=
\angsem{\hat I}{Q^{\ge 1}_1}(a)\ \ominus\ \angsem{\hat I}{Q^{\ge (C+1)}_1}(a).
\]
By the $\ge$-case, $\angsem{\hat I}{Q^{\ge 1}_1}(a)=S_1([|U^{\hat I}_{q,(a)}|])$ and
$\angsem{\hat I}{Q^{\ge (C+1)}_1}(a)=S_{C+1}([|U^{\hat I}_{q,(a)}|])$. Hence
$
\angsem{\hat I}{Q^{\le C}_1}(a)
=
S_1([|U^{\hat I}_{q,(a)}|])\ominus S_{C+1}([|U^{\hat I}_{q,(a)}|])
=
\angsem{\hat I}{Q^{\le C}_2}(a)
$ by $(J_{=,\le})$.

\smallskip
Since group key $(a)$ was arbitrary, we conclude
$\angsem{\hat I}{Q^{\op C}_1}=\angsem{\hat I}{Q^{\op C}_2}$ for
$\op\in\{\ge,=,\le\}$.
\paragraph*{Proof of (ii)}
\lean{Semirings/Why}{Why.counterexample_having}
We show that there exists a $\mathbb{K}$-instance $\hat I$ such that
$\angsem{\hat I}{Q^{\op C}_1}\neq \angsem{\hat I}{Q^{\op C}_2}$.
Fix $C=2$ and $\op= \; =$.
Since these queries are equivalent, both of them return groups which have 2 tuples. Say, in a certain $\hat I$ the result of the pre-aggregation subquery $q$ can form a valid group with group-key $u$ with $\mathbb K$-annotations $t_1,t_2,\text{ and }t_3$.
Then,
\[
  \angsem{\hat I}{Q^{= 2}_1}=(t_1 \otimes t_2)\otimes(\mathbb{1}\ominus t_3) \oplus (t_1 \otimes t_3)\otimes(\mathbb{1}\ominus t_2)\oplus(t_2 \otimes t_3)\otimes(\mathbb{1}\ominus t_1)
\]
and,
\[
  \angsem{\hat I}{Q^{= 2}_2}=(t_1 \otimes t_2)\oplus (t_1 \otimes t_3)\oplus (t_2 \otimes t_3)
\]
Fix a Why-semiring valuation of $t_1, t_2, t_3$: $\{\{a\}\}, \{\{b\}\}, \{\emptyset\}$.
So,
\begin{align*}
  &\angsem{\hat I}{Q^{= 2}_1}(u)=\\
  &((\{\{a\}\}\Cup \{\{b\}\})\Cup(\{\emptyset\}\setminus \{\emptyset\})) \cup ((\{\{a\}\}\Cup \{\emptyset\})\Cup(\{\emptyset\}\setminus \{\{b\}\})) \\
  &\cup((\{\{b\}\}\Cup \{\emptyset\})\Cup(\{\emptyset\}\setminus \{\{a\}\}))\\
  =&(\{\{a\}\}\Cup \{\{b\}\}\Cup \emptyset) \cup (\{\{a\}\}\Cup \{\emptyset\}) \cup (\{\{b\}\}\Cup \{\emptyset\})\\
  =&\{\{a\},\{b\}\}
\end{align*}

But,
\begin{align*}
  &\angsem{\hat I}{Q^{= 2}_2}(u)=\\
  &(\{\{a\}\}\Cup\{\{b\}\})\cup(\{\{a\}\}\Cup\{\emptyset\})\cup(\{\{b\}\}\Cup\{\emptyset\})\\
  =&\{\{a, b\}\} \cup \{\{a\}\} \cup \{\{b\}\}\\
  =&\{\{a\}, \{b\}, \{a, b\}\}
\end{align*}
Hence, $\angsem{\hat I}{Q^{\op C}_1}\neq \angsem{\hat I}{Q^{\op C}_2}$.

\begin{proofsketch}

Hence, it suffices to show that for each group produced by the grouping operator in $Q^{\op C}_1$, the evaluated $\mathbb K$-provenance because of the aggregate comparison coincides with that produced by $Q^{\op C}_2$ for the corresponding tuple in $\angsem{\hat I}{Q^{\op C}_2}$.
Since $q$ is an arity~$2$ query, every tuple in $\angsem{\hat I}{q}$ has the form $(u,\alpha)$ where u is $(u_1,u_2)$ and $\alpha\in \mathbb K$. We fix an arbitrary group-key $\vec v=(\#1(u)=a)$. So,

$\angsem{\hat I}{Q^{\op C}_1}(a)=\predsem{\hat I}{\gamma_{\#1}[1:+](q)}{\#2\op C}(u)$ is
$\bigoplus_{\substack{W\subseteq [|U^{\hat I}_{q,(a)}|],\\ W\neq\emptyset,\\|W|\op C}} T_{[|U^{\hat I}_{q,(a)}|]}(W)$ where,

is indeed the $\ann^{\hat I}_{q,(a)}(W)$ for each possible-world $W\subseteq [|U^{\hat I}_{q,(a)}|]$,
$A_W$ is $\bigotimes_{w\in W}(U^{\hat I}_{q,(a)})_w[\alpha]=\bigotimes_{w\in W}\alpha_w$ and
$B_W$ is $\bigoplus_{w\in [|U^{\hat I}_{q,(a)}|]\setminus W}(U^{\hat I}_{q,(a)})_w[\alpha] =\bigoplus_{w\in [|U^{\hat I}_{q,(a)}|]\setminus W}\alpha_w$.
For every $W\subseteq [|U^{\hat I}_{q,(a)}|] $ and $C\ge 1$ we define

$\angsem{\hat I}{Q^{\ge C}_2}(a)=S_C([|U^{\hat I}_{q,(a)}|])$.\\
A join-valuation contributing to the output
tuple in $\angsem{\hat I}{Q^{\ge C}_2}$  with key $(a)$ corresponds to choosing $C$ tuples from $\angsem{\hat I}{q}$ with first
attribute $a$ and strictly increasing second attributes (due to the chain of
join conditions $\#2<^*\#4<^*\cdots$) with a tie-breaking occurence attribute as in $U^{\hat I}_{q,(a)}$ so that it is possible to select distinct tuple occurences with equal second attributes in an increasing order. Thus, complying with multiset semantics, this corresponds to choosing a
$C$-element subset $W\subseteq[|U^{\hat I}_{q,(a)}|]$, and
$\Pi_{\#1}$ then takes the $\oplus$-sum over all these subsets.\\
By definition, $Q^{=C}_2 = Q^{\ge C}_2 \ominus Q^{\ge (C+1)}_2$ and $Q^{\le C}_2 = \Pi_{\#1}(q) \ominus Q^{\ge (C+1)}_2$.
Since, $\angsem{\hat I}{Q^{\ge C}_2}(a)=S_C([|U^{\hat I}_{q,(a)}|])$, we write:
\[
\angsem{\hat I}{Q^{=C}_2}(a)
=
S_C([|U^{\hat I}_{q,(a)}|]) \ominus S_{C+1}([|U^{\hat I}_{q,(a)}|])
\text{ and }
\angsem{\hat I}{Q^{\le C}_2}(a)
=
S_1([|U^{\hat I}_{q,(a)}|]) \ominus S_{C+1}([|U^{\hat I}_{q,(a)}|]).
\]
We now claim a weak bound on $\mathbb K$-element(s) implying joint tuple-presence in corresponding \emph{possible-worlds}.

\begin{lemmarep}\label{lem:upward-expansion}
\lean{Having}{Having.upward_expansion}
Let $(K,\oplus,\otimes,\mathbb 0,\mathbb 1,\ominus)$ be an idempotent m-semiring. Fix a finite set $U$ and elements $(\alpha)_{x\in U}\in \mathbb K$. For $W\subseteq U$ define
$A_W := \bigotimes_{x\in W}\alpha_x$ and $T_U(W) := A_W \ominus \bigoplus_{x\in U\setminus W} A_{W\cup\{x\}}$.
Then for every $V\subseteq U$, $A_V \;\le\; \bigoplus_{\substack{W:\;V\subseteq W\subseteq U}} T_U(W)$.
\end{lemmarep}

\begin{appendixproof}
From the properties of $m$-semirings $\ominus$ is the (right) residual of $\oplus$, i.e.\ for all $x,y,z\in K$,
\begin{equation}\label{eq:residuation}
  x\ominus y \le z \quad\Longleftrightarrow\quad x \le y\oplus z .
\end{equation}

We prove
\begin{equation}\label{eq:upward-expansion}
  A_V \;\le\; \bigoplus_{\substack{W:\;V\subseteq W\subseteq U}} T_U(W).
\end{equation}
by induction on $d:=|U\setminus V|$.

\smallskip\noindent
\textbf{Base case ($d=0$).}
Then $V=U$. Since $U\setminus U=\varnothing$ and $\bigoplus_{\varnothing}(\cdot)=\mathbb 0$,
\[
  T_U(U)=A_U\ominus \mathbb 0.
\]
We claim $A_U\ominus\mathbb 0=A_U$. Indeed, by \eqref{eq:residuation}, for any $z\in K$,
\[
  A_U\ominus\mathbb 0 \le z \iff A_U \le \mathbb 0\oplus z = z,
\]
so $A_U\ominus\mathbb 0$ is the greatest element below $A_U$, hence equals $A_U$.
Therefore,
\[
  \bigoplus_{\substack{W:\;U\subseteq W\subseteq U}} T_U(W)=T_U(U)=A_U=A_V,
\]
and \eqref{eq:upward-expansion} holds.

\smallskip\noindent
\textbf{Inductive step.}
Assume $d>0$ and that \eqref{eq:upward-expansion} holds for all pairs $(U,V')$
with $V\subsetneq V'\subseteq U$ and $|U\setminus V'|<d$.
Let
\[
  Y := \bigoplus_{x\in U\setminus V} A_{V\cup\{x\}} \in K.
\]
From Lemma~\ref{lem:x<=y+(x-y)} we obtain the standard inequality
\begin{equation}\label{eq:M2-lemma1}
  X \le Y \oplus (X\ominus Y)\qquad\text{for all }X,Y\in K,
\end{equation}
by instantiating \eqref{eq:residuation} with $z=X\ominus Y$ (since $X\ominus Y\le X\ominus Y$).
Applying \eqref{eq:M2-lemma1} to $X=A_V$ gives
\[
  A_V \le Y \oplus (A_V\ominus Y).
\]
By definition of $T_U(V)$ we have $A_V\ominus Y = T_U(V)$, hence
\begin{equation}\label{eq:AV-bound}
  A_V \le T_U(V)\ \oplus\ \bigoplus_{x\in U\setminus V} A_{V\cup\{x\}}.
\end{equation}

For each $x\in U\setminus V$ set $V_x:=V\cup\{x\}$. Then $|U\setminus V_x|=d-1$,
so the induction hypothesis yields
\[
  A_{V_x}\le \bigoplus_{\substack{W:\;V_x\subseteq W\subseteq U}} T_U(W).
\]
Taking the $\oplus$-sum over all $x\in U\setminus V$ and using monotonicity of $\oplus$,
\[
  \bigoplus_{x\in U\setminus V} A_{V\cup\{x\}}
  =
  \bigoplus_{x\in U\setminus V} A_{V_x}
  \le
  \bigoplus_{x\in U\setminus V}\ \bigoplus_{\substack{W:\;V_x\subseteq W\subseteq U}} T_U(W).
\]
Every $W$ appearing on the right satisfies $V\subseteq V_x\subseteq W\subseteq U$, hence $V\subseteq W\subseteq U$.
Therefore, using associativity/commutativity/idempotence of $\oplus$ to ignore duplicates and reorder summands,
\[
  \bigoplus_{x\in U\setminus V}\ \bigoplus_{\substack{W:\;V_x\subseteq W\subseteq U}} T_U(W)
  \le
  \bigoplus_{\substack{W:\;V\subseteq W\subseteq U}} T_U(W).
\]
Combining with \eqref{eq:AV-bound} yields
\[
  A_V \le T_U(V)\ \oplus\ \bigoplus_{\substack{W:\;V\subseteq W\subseteq U}} T_U(W)
  =
  \bigoplus_{\substack{W:\;V\subseteq W\subseteq U}} T_U(W),
\]
since the right-hand $\oplus$-sum already includes the summand $T_U(V)$ (take $W=V$).
This proves \eqref{eq:upward-expansion} for $V$ and completes the induction.
\end{appendixproof}

\begin{lemmarep}[Recurrence for $F_C$]\label{lem:FC_recurrence}
\lean{Having}{Having.FC_recurrence}
Let $(\mathbb K,\oplus,\otimes,\mathbb 0,\mathbb 1,\ominus)$ be an idempotent m-semiring satisfying distributivity of $\otimes$ over $\ominus$. Fix a non-empty finite set $U$ and elements $(\alpha)_{x\in U}\in \mathbb K$. For $W\neq\emptyset\subseteq U$ define
$A_W:=\bigotimes_{x\in W}\alpha_x$ and $T_U(W):=A_W\ominus \bigoplus_{x\in U\setminus W}A_{W\cup\{x\}}$.
Further for $C\in\mathbb N^{\geq 1}$ define
$
F_C(U):=\bigoplus_{\substack{W\subseteq U\\|W|\ge C}} T_U(W).
$
Then for an arbitrary choice of $u\in U$ such that $U'=U\setminus\{u\}$ $F_C(U)=F_C(U')\ \oplus\ \bigl(F_{C-1}(U')\otimes\alpha_u\bigr)$.
\end{lemmarep}

\begin{appendixproof}
  Fix $u\in U$ such that $U'=U\setminus\{u\}$ and a $C\in\mathbb N^{\geq 1}$.
Split  $F_C(U)$ according to whether $u\in W$:
\begin{equation}\label{eq:split-I-II}
F_C(U)
=
\underbrace{\bigoplus_{\substack{W\subseteq U'\\|W|\ge C}}T_U(W)}_{(I).\;u\notin W}
\ \oplus\
\underbrace{\bigoplus_{\substack{W'\subseteq U'\\|W'|\ge C-1}}T_U(W'\cup\{u\})}_{(II).\;u\in W}.
\end{equation}

\paragraph{Simplify $(II)$: }
Fix $W'\subseteq U'$. Then $A_{W'\cup\{u\}}=A_{W'}\otimes\alpha_u$, and
$U\setminus(W'\cup\{u\})=U'\setminus W'$. Hence, using distributivity of $\otimes$ over $\oplus$,
\[
\bigoplus_{x\in U\setminus(W'\cup\{u\})}A_{W'\cup\{u,x\}}
=
\bigoplus_{x\in U'\setminus W'}\bigl(A_{W'\cup\{x\}}\otimes\alpha_u\bigr)
=
\Bigl(\bigoplus_{x\in U'\setminus W'}A_{W'\cup\{x\}}\Bigr)\otimes\alpha_u.
\]
Therefore, using distributivity of $\otimes$ over $\ominus$,
\[
T_U(W'\cup\{u\})
=
(A_{W'}\otimes\alpha_u)\ominus
\Bigl(\bigoplus_{x\in U'\setminus W'}A_{W'\cup\{x\}}\Bigr)\otimes\alpha_u
=
\Bigl(A_{W'}\ominus\bigoplus_{x\in U'\setminus W'}A_{W'\cup\{x\}}\Bigr)\otimes\alpha_u
=
T_{U'}(W')\otimes\alpha_u.
\]
Taking $\oplus$ over all $W'\subseteq U'$ with $|W'|\ge C-1$ and factoring $\alpha_u$ using
distributivity of $\otimes$ over $\oplus$ yields
\begin{equation}\label{eq:II-simplifies}
(II)=\Bigl(\bigoplus_{\substack{W'\subseteq U'\\|W'|\ge C-1}}T_{U'}(W')\Bigr)\otimes\alpha_u
=F_{C-1}(U')\otimes\alpha_u.
\end{equation}

\paragraph{Relate $T_U(W)$ and $T_{U'}(W)$ for $W\subseteq U'$: }
Fix $W\subseteq U'$. Since $U\setminus W=(U'\setminus W)\cup\{u\}$, we have
\[
\bigoplus_{x\in U\setminus W}A_{W\cup\{x\}}
=
\Bigl(\bigoplus_{x\in U'\setminus W}A_{W\cup\{x\}}\Bigr)\oplus (A_W\otimes\alpha_u).
\]
Using
$x\ominus(y\oplus z)=(x\ominus y)\ominus z$, we get
\begin{equation}\label{eq:TU-rewrite}
T_U(W)
=
A_W\ominus\Bigl(\bigoplus_{x\in U'\setminus W}A_{W\cup\{x\}}\oplus(A_W\otimes\alpha_u)\Bigr)
=
T_{U'}(W)\ominus(A_W\otimes\alpha_u).
\end{equation}

\paragraph{Show $F_C(U)\le F_C(U')\oplus(F_{C-1}(U')\otimes\alpha_u)$: }
  By \eqref{eq:TU-rewrite} and the inequality $x\ominus y\le x$ (shown in Lemma~\ref{lem:x-y<=x}),
we have $T_U(W)\le T_{U'}(W)$ for all $W\subseteq U'$.
Taking $\oplus$ over all $W\subseteq U'$ with $|W|\ge C$ yields $(I)\le F_C(U')$.
Together with \eqref{eq:split-I-II} and \eqref{eq:II-simplifies}, this gives
\begin{equation}\label{eq:leq-direction}
F_C(U)\le F_C(U')\oplus(F_{C-1}(U')\otimes\alpha_u).
\end{equation}

\paragraph{Show $F_C(U')\le F_C(U)$: }
Fix $W\subseteq U'$ with $|W|\ge C$.
By the inequality $x\le y\oplus(x\ominus y)$ (shown in Lemma~\ref{lem:x<=y+(x-y)}) applied to
$x=T_{U'}(W)$ and $y=A_W\otimes\alpha_u$, and using \eqref{eq:TU-rewrite}, we get
\begin{equation}\label{eq:TUprime-bound}
T_{U'}(W)\le (A_W\otimes\alpha_u)\oplus T_U(W).
\end{equation}
Since $|W|\ge C$, the term $T_U(W)$ is one of the summands of $F_C(U)$, hence $T_U(W)\le F_C(U)$.
Moreover $A_W\otimes\alpha_u=A_{W\cup\{u\}}$, and $|W\cup\{u\}|\ge C$.
By Lemma~\ref{lem:upward-expansion} applied to $V=W\cup\{u\}$ we have
\[
A_{W\cup\{u\}}
\le
\bigoplus_{\substack{Y:\;W\cup\{u\}\subseteq Y\subseteq U}} T_U(Y)
\le
\bigoplus_{\substack{Y\subseteq U\\|Y|\ge C}}T_U(Y)
=
F_C(U),
\]
so $A_W\otimes\alpha_u\le F_C(U)$.
Plugging these two bounds into \eqref{eq:TUprime-bound} yields $T_{U'}(W)\le F_C(U)$.
Taking $\oplus$ over all $W\subseteq U'$ with $|W|\ge C$ gives $F_C(U')\le F_C(U)$.

\paragraph{The recurrence: }
From \eqref{eq:leq-direction}, $F_C(U')\le F_C(U)$ and
$F_{C-1}(U')\otimes\alpha_u=(II)\le F_C(U)$, we obtain
$F_C(U')\oplus(F_{C-1}(U')\otimes\alpha_u)\le F_C(U)$.
Together with \eqref{eq:leq-direction}, we get
\[
F_C(U)=F_C(U')\oplus(F_{C-1}(U')\otimes\alpha_u),
\]
as required.
\end{appendixproof}
\begin{lemmarep}[Recurrence for $S_C$]\label{lem:SC_recurrence}
\lean{Having}{Having.SC_recurrence}
Let $(\mathbb K,\oplus,\otimes,\mathbb 0,\mathbb 1,\ominus)$ be an m-semiring.
Fix a non-empty finite set $U$ and a family $(\alpha)_{x\in U}\in \mathbb K$.
For $W\neq\emptyset\subseteq U$ define $A_W:=\bigotimes_{x\in W}\alpha_x$ and for $C\in\mathbb N^{\ge 1}$ define
$
S_C(U):=\bigoplus_{\substack{W\subseteq U\\|W|=C}} A_W$.
For an arbitrary choice of $u\in U$ such that $U'=U\setminus\{u\}$ for every $C\ge 1$,
$
S_C(U)=S_C(U')\ \oplus\ \bigl(S_{C-1}(U')\otimes\alpha_u\bigr).
$
\end{lemmarep}

\begin{appendixproof}
Fix $C\ge 1$ and $u\in U$, and write $U=U'\cup\{u\}$ with $U'=U\setminus\{u\}$.

Define two index families:
\[
\mathcal A:=\{\,W\subseteq U \mid |W|=C,\ u\notin W\,\},
\qquad
\mathcal B:=\{\,W\subseteq U \mid |W|=C,\ u\in W\,\}.
\]
Then $\mathcal A$ and $\mathcal B$ are disjoint and their union is the full family
\[
\mathcal A\ \dot\cup\ \mathcal B=\{\,W\subseteq U\mid |W|=C\,\},
\]
because every $W\subseteq U$ either contains $u$ or does not.

Therefore, by associativity and commutativity of $\oplus$ (finite sum over a disjoint
union equals the $\oplus$ of the two partial sums),
\begin{equation}\label{eq:split-SC}
S_C(U)
=
\bigoplus_{W\in\mathcal A} A_W\ \oplus\ \bigoplus_{W\in\mathcal B} A_W.
\end{equation}

\smallskip\noindent
\emph{(i) The $\mathcal A$-part.}
If $W\in\mathcal A$, then $u\notin W$ and $W\subseteq U'=U\setminus\{u\}$.
Conversely, if $W\subseteq U'$ and $|W|=C$, then $W\subseteq U$ and $u\notin W$.
Hence
\[
\mathcal A=\{\,W\subseteq U'\mid |W|=C\,\},
\]
so
\begin{equation}\label{eq:A-part}
\bigoplus_{W\in\mathcal A}A_W
=
\bigoplus_{\substack{W\subseteq U'\\|W|=C}}A_W
=
S_C(U').
\end{equation}

\smallskip\noindent
\emph{(ii) The $\mathcal B$-part.}
Define a map
\[
\phi:\{\,W'\subseteq U'\mid |W'|=C-1\,\}\longrightarrow \mathcal B,
\qquad
\phi(W'):=W'\cup\{u\}.
\]
This map is a bijection:
\begin{itemize}
\item It is well-defined because if $W'\subseteq U'$ and $|W'|=C-1$, then
$u\notin W'$ and thus $|W'\cup\{u\}|=|W'|+1=C$, and $\phi(W')$ contains $u$.
\item It is injective because if $W'_1\cup\{u\}=W'_2\cup\{u\}$, removing $u$ gives
$W'_1=W'_2$.
\item It is surjective because if $W\in\mathcal B$, then $u\in W$ and setting
$W':=W\setminus\{u\}$ gives $W'\subseteq U'$ and $|W'|=C-1$, with $W=\phi(W')$.
\end{itemize}
Hence we may re-index the sum over $\mathcal B$ via $\phi$:
\begin{equation}\label{eq:B-reindex}
\bigoplus_{W\in\mathcal B}A_W
=
\bigoplus_{\substack{W'\subseteq U'\\|W'|=C-1}} A_{\phi(W')}
=
\bigoplus_{\substack{W'\subseteq U'\\|W'|=C-1}} A_{W'\cup\{u\}}.
\end{equation}
For each $W'\subseteq U'$ we have, by commutativity and associativity of $\otimes$,
\[
A_{W'\cup\{u\}}
=
\Bigl(\bigotimes_{x\in W'}\alpha_x \Bigr)\otimes \alpha_u
=
A_{W'}\otimes \alpha_u.
\]
Substituting into \eqref{eq:B-reindex} and using distributivity of $\otimes$ over $\oplus$,
\begin{equation}\label{eq:B-part}
\bigoplus_{W\in\mathcal B}A_W
=
\bigoplus_{\substack{W'\subseteq U'\\|W'|=C-1}} (A_{W'}\otimes\alpha_u)
=
\Bigl(\bigoplus_{\substack{W'\subseteq U'\\|W'|=C-1}} A_{W'}\Bigr)\otimes\alpha_u
=
S_{C-1}(U')\otimes\alpha_u.
\end{equation}

Finally, combining \eqref{eq:split-SC}, \eqref{eq:A-part}, and \eqref{eq:B-part}, we obtain
\[
S_C(U)=S_C(U')\ \oplus\ \bigl(S_{C-1}(U')\otimes\alpha_u\bigr),
\]
as required.
\end{appendixproof}

  $\mathbb{0}_\mathbb{K}$ is NP-hard.

  Furthermore, this problem is in NP for semirings such as
  $\mathbb{N}$ or $\mathbb{B}[X]$.
\end{proposition}

  $\mathbb{K}$ (keeping integers in $\mathbb{N}$ and unexpanded circuits for
  the Boolean expressions in $\mathbb{B}[X]$) using the standard
  semantics of relational algebra operators in $\mathbb{K}$-semirings
  \cite{sen2026provsql}.

  Definition~\ref{def:semantics}:

\bigoplus_{\substack{W\subseteq [|U^{\hat I}_{R,()}|]\\ W\neq\emptyset}}
\Bigl(
  \chi_{\op}\bigl(\Sigma_{j\in W} m_{i_j}, B)\bigr)
\Bigr),
\]
where $\omega$ is any arbitrary ordering since our aggregate here is commutative.
This is non-zero if and only if one of the term of the $\oplus$ sum is non-zero since
$\mathbb K$ has characteristic 0, i.e., if the subset-sum instance has a
solution.
\end{proof}
