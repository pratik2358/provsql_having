In this section we describe specialized procedures for evaluating the valid possible-worlds for selections involving aggregate comparisons.

ProvSQL implements multiset semantics, using \emph{tuple occurrences} carrying globally unique provenance identifiers (UUIDs).
We maintain a uniqueness invariant for every intermediate query result $\angsem{\hat I}{q}$: no pair $(u,\alpha)$ occurs twice, meaning that duplicates of the same value-tuple $u$ are represented as distinct occurrences with different annotations $\alpha$.
Under this invariant, the multiset of group member occurrences can be treated as a finite \emph{set}, and possible worlds are then ordinary subsets.
This does not change our semantics: multiset properties are preserved because distinct occurrences may still yield equal values $t(u)$, so the induced aggregate input $\vals_t(W)=\mset{t(u)\mid (u,\alpha)\in W}$ remains a multiset of values.

All our algorithms are applied \emph{per group key} $\vec v$ and operate on the set of group member occurrences $U^{\hat I}_{q',\vec v}$ of the \emph{pre-aggregation} subquery $q'$.
Accordingly, any term $t$ handled by these procedures is evaluated on tuples produced by $q'$, and its max-index is bounded by the arity of $q'$ (denoted by $k$ in the algorithms for readability).

For bounded range comparisons of the form $\mathrm{SUM}(t)\op C$, a dynamic programming procedure is presented in Algorithm~\ref{alg:sum_dp}.
It enumerates the possible worlds $W\subseteq U^{\hat I}_{q',\vec v}$ whose sum $\sum_{(u,\alpha)\in W} t(u)$(i.e. $\SUM(t)$) satisfies the comparison with the constant $C$, i.e., precisely the worlds contributing nonzero weight to the corresponding $\chi_{\op}$-ensured possible-world sum in the semantics of aggregate comparison atoms.

The algorithm works also for the special case where $t=1$, i.e., bounded range $\mathrm{COUNT}(t)\op C$ comparisons.
However, for this case we can use a more efficient enumeration, exploiting the unit-weight sum structure of $\mathrm{COUNT}$ aggregates, as shown in Algorithm~\ref{alg:count_enum}.
Similarly, for $\mathrm{MIN}$/$\mathrm{MAX}$ aggregates we can use strategies based on ordering the possible aggregate values that avoid full world enumeration, and only listing the worlds that are under or over the threshold $C$ as needed.

For aggregate functions or predicates not covered by the specialized cases, reverting to
brute-force enumeration of non-empty worlds $W\subseteq U^{\hat I}_{q',\vec v}$ will always give the correct answer by our semantics.
The algorithms can also be used to evaluate provenance for comparisons between two aggregates, by evaluating one of the aggregates into a constant value first. \sil{Do we need to list all possible constants for the second aggregate?}

\begin{algorithm}
  \caption{Enumeration of valid worlds for $(t:\mathrm{SUM}) \op C$ at a fixed group key $\vec v$}
  \label{alg:sum_dp}

  \Input{a finite set of group-member occurrences $U^{\hat I}_{q',\vec v}$, a term $t$ of max-index $\le k$ such that $t(u)\in\NN \; \forall (u,\alpha)\in U^{\hat I}_{q',\vec v}$ , a constant $C\in\NN$, and an operator $\op\in\{=,>,<,\ge,\le,\neq\}$}
  \Output{a list $\mathcal W$ of worlds $\emptyset\neq W\subseteq U^{\hat I}_{q',\vec v}$ such that $\sum_{(u,\alpha)\in W} t(u)\ \op\ C$}
  \BlankLine
  \Fn{\textsc{SumDP}$(U^{\hat I}_{q',\vec v},t,C,\op)$}{
  \If{$\op = \neq$}{
    \Return $\textsc{SumDP}(U^{\hat I}_{q',\vec v},t,C,<) \cup \textsc{SumDP}(U^{\hat I}_{q',\vec v},t,C,>)$\;
  }
  \BlankLine
  Represent $U^{\hat I}_{q',\vec v}$ as a finite list of distinct occurrences $\big((u_i,\alpha_i)\big)_{1\le i\le N}$\;
  $T \gets \sum_{i=1}^{N} t(u_i)$\;
  \If{$(\op = = \textbf{ and } C > T)$ \textbf{or} $(\op = > \textbf{ and } C \ge T)$ \textbf{or} $(\op = \ge \textbf{ and } C > T)$ \textbf{or} $(\op = < \textbf{ and } C = 0)$}{
    \Return $\emptyset$\;
  }
  \If{$\op\in\{>,\ge\}$}{
    $J \gets T$\;
  }
  \ElseIf{$\op=<$}{
    $J \gets \min(C-1,\ T)$\;
  }
  \Else{
    $J \gets \min(C,\ T)$\;
  }

  \BlankLine
  Initialize $dp[0] \leftarrow \{\emptyset\}$\;
  \For{$j \gets 1$ \KwTo $J$}{
    $dp[j] \leftarrow \emptyset$\;
  }

  \BlankLine
  $\mathrm{pref\_sum} \gets 0$\;

  \For{$i \gets 1$ \KwTo $N$}{
    $w \gets t(u_i)$\;
    $\mathrm{pref\_sum} \gets \mathrm{pref\_sum} + w$\;
    $j_{\max} \gets \min(J,\ \mathrm{pref\_sum})$\;

    \For{$j \gets j_{\max}, j_{\max}-1, \dots, w$}{
      $p \gets j - w$\;
      \If{$dp[p] \neq \emptyset$}{
        let $s \gets |dp[p]|$\;
        \For{$r \gets 1$ \KwTo $s$}{
          append $dp[p][r] \cup \{(u_i,\alpha_i)\}$ to $dp[j]$\;
        }
      }
    }
  }

  \BlankLine
  \If{$\op = =$}{
    $\mathcal W \leftarrow dp[C]$\;
  }
  \ElseIf{$\op = >$}{
    $\mathcal W \leftarrow \bigcup_{j=C+1}^{J} dp[j]$\;
  }
  \ElseIf{$\op = <$}{
    $\mathcal W \leftarrow \bigcup_{j=0}^{C-1} dp[j]$\;
  }
  \ElseIf{$\op = \ge$}{
    $\mathcal W \leftarrow \bigcup_{j=C}^{J} dp[j]$\;
  }
  \Else{
    $\mathcal W \leftarrow \bigcup_{j=0}^{C} dp[j]$\;
  }

  $\mathcal W \leftarrow \mathcal W \setminus \{\emptyset\}$\;

  \Return $\mathcal W$\;
  }
\end{algorithm}

Both algorithms are correct by the following two theorems.
\begin{theoremrep}[Correctness of Algorithm~\ref{alg:sum_dp}]\label{thm:sumdp-correct}
Fix a $\mathbb K$-instance $\hat I$ and a group key $\vec v$.
Let $U\defeq U^{\hat I}_{q',\vec v}$ and represent it as a finite list of distinct occurrences
$((u_i,\alpha_i))_{1\le i\le N}$.
Let $t$ be a term of max-index $\le k$ such that $t(u_i)\in\NN$ for all $1\le i\le N$.
Fix $C\in\NN$ and $\op\in\{=,>,<,\ge,\le,\neq\}$.
Let $\mathcal W$ be the output of Algorithm~\ref{alg:sum_dp} on input $(U,t,C,\op)$.
Then
\[
\mathcal W
=
\{\,W\subseteq U\mid W\neq\emptyset,\sum_{(u,\alpha)\in W}t(u)\op C\,\}.
\]
\end{theoremrep}

\begin{appendixproof}
Write
\[
T\defeq \sum_{i=1}^N t(u_i).
\]

\smallskip\noindent\textbf{Case $\op=\neq$.}
If $\op=\neq$, the algorithm returns
\[
\textsc{SumDP}(U,t,C,<)\cup\textsc{SumDP}(U,t,C,>).
\]
By the correctness proved below for $\op\in\{<,>\}$, this equals
\[
\{\,W\subseteq U\mid W\neq\emptyset,\sum_{(u,\alpha)\in W}t(u)<C\,\}
\cup
\{\,W\subseteq U\mid W\neq\emptyset,\sum_{(u,\alpha)\in W}t(u)>C\,\}.
\]
Since for every integer $x$ one has $x\neq C$ iff $x<C$ or $x>C$, the latter union is exactly
\[
\{\,W\subseteq U\mid W\neq\emptyset,\sum_{(u,\alpha)\in W}t(u)\neq C\,\}.
\]
Hence the theorem holds for $\op=\neq$.

\smallskip\noindent
The remainder of the proof establishes correctness for $\op\in\{=,>,<,\ge,\le\}$.

\smallskip\noindent\textbf{Early-return cases.}
The algorithm returns $\emptyset$ in the following scenarios:
\begin{itemize}
\item if $\op$ is $=$ and $C>T$;
\item if $\op$ is $>$ and $C\ge T$;
\item if $\op$ is $\ge$ and $C>T$;
\item if $\op$ is $<$ and $C=0$.
\end{itemize}
If $\op$ is $=$ and $C>T$, then every world $W\subseteq U$ satisfies
\[
\sum_{(u,\alpha)\in W} t(u)\le \sum_{i=1}^N t(u_i)=T<C,
\]
so no world has sum $C$.

If $\op$ is $>$ and $C\ge T$, then every world $W\subseteq U$ satisfies
\[
\sum_{(u,\alpha)\in W} t(u)\le T\le C,
\]
so no world has sum strictly greater than $C$.

If $\op$ is $\ge$ and $C>T$, then again every world has sum at most $T<C$, so no world has sum at least $C$.

If $\op$ is $<$ and $C=0$, since all values $t(u_i)$ lie in $\NN$, every world has nonnegative sum; hence no world has sum $<0$.

Therefore, in each early-return case, the algorithm correctly outputs $\emptyset$.

\smallskip\noindent\textbf{DP invariant.}
We can safely assume from now on that none of the early-return cases occur.
For $0\le i\le N$, let
\[
U_i\defeq \{(u_1,\alpha_1),\dots,(u_i,\alpha_i)\},
\]
with $U_0=\emptyset$.
For $0\le i\le N$ and $j\ge 0$, define
\[
\mathcal D_i(j)\defeq
\{\,W\subseteq U_i\mid \sum_{(u,\alpha)\in W} t(u)=j\,\}.
\]
We claim that after completing the outer-loop iteration with index $i$,
for every $j\in\{0,\dots,J\}$,
\begin{equation}\label{eq:sumdp-invariant}
dp[j]=\mathcal D_i(j).
\end{equation}

We prove this by induction on $i$.

\smallskip\noindent\emph{Base case $i=0$.}
The algorithm initializes
\[
dp[0]=\{\emptyset\}
\qquad\text{and}\qquad
dp[j]=\emptyset \text{ for } 1\le j\le J.
\]
Since $U_0=\emptyset$, the only subset of $U_0$ is $\emptyset$, and its sum is $0$.
Thus
\[
\mathcal D_0(0)=\{\emptyset\}
\qquad\text{and}\qquad
\mathcal D_0(j)=\emptyset \text{ for } j\ge 1.
\]
Hence \eqref{eq:sumdp-invariant} holds for $i=0$.

\smallskip\noindent\emph{Inductive step.}
Assume \eqref{eq:sumdp-invariant} holds after iteration $i-1$, for some $1\le i\le N$.
Fix $j\in\{0,\dots,J\}$.
We show that after iteration $i$, one has $dp[j]=\mathcal D_i(j)$.

We now establish two cases.

\smallskip\noindent\underline{Case 1: $j<t(u_i)$.}
The inner loop does not update $dp[j]$, since the loop runs only for indices
\[
j=j_{\max},j_{\max}-1,\dots,t(u_i).
\]
Hence after iteration $i$, the value of $dp[j]$ is unchanged, so by the induction hypothesis
\[
dp[j]=\mathcal D_{i-1}(j).
\]
Further, no subset of $U_i$ containing $(u_i,\alpha_i)$ can have sum $j$, because such a subset has sum at least
\[
t(u_i)>j,
\]
and all summands are nonnegative. Therefore every world in $\mathcal D_i(j)$ must avoid $(u_i,\alpha_i)$, so
\[
\mathcal D_i(j)=\mathcal D_{i-1}(j).
\]
Consequently,
\[
dp[j]=\mathcal D_i(j).
\]

\smallskip\noindent\underline{Case 2: $j\ge t(u_i)$.}
A subset $W\subseteq U_i$ satisfies
\[
\sum_{(u,\alpha)\in W} t(u)=j
\]
iff exactly one of the following two disjoint possibilities holds:
\begin{itemize}
\item $(u_i,\alpha_i)\notin W$, in which case $W\subseteq U_{i-1}$ and
\[
W\in \mathcal D_{i-1}(j);
\]
\item $(u_i,\alpha_i)\in W$, in which case
\[
W=W'\cup\{(u_i,\alpha_i)\}
\]
for a unique subset $W'\subseteq U_{i-1}$ satisfying
\[
\sum_{(u,\alpha)\in W'} t(u)=j-t(u_i),
\]
that is,
\[
W'\in \mathcal D_{i-1}(j-t(u_i)).
\]
\end{itemize}

By the induction hypothesis, before processing occurrence $i$ we have
\[
dp[j]=\mathcal D_{i-1}(j)
\qquad\text{and}\qquad
dp[j-t(u_i)]=\mathcal D_{i-1}(j-t(u_i)).
\]
During the update for this fixed $j$, the algorithm sets
\[
p\gets j-t(u_i),
\qquad
s\gets |dp[p]|.
\]
It then appends
\[
dp[p][r]\cup\{(u_i,\alpha_i)\}
\qquad (1\le r\le s)
\]
to $dp[j]$.

Thus the worlds added to $dp[j]$ are exactly those obtained by adding $(u_i,\alpha_i)$ to the worlds that were already present in $dp[p]$ before any append operation at this $j$.
Since, by the I.H, those pre-existing worlds are exactly the elements of $\mathcal D_{i-1}(j-t(u_i))$, the algorithm adds exactly the worlds in
\[
\{\,W'\cup\{(u_i,\alpha_i)\}\mid W'\in \mathcal D_{i-1}(j-t(u_i))\,\}.
\]

$s=|dp[p]|$ is essential when $t(u_i)=0$, for then $p=j$.
In that situation the algorithm appends new worlds to $dp[j]$ while scanning from $dp[j]$, but it scans only the first $s$ entries, i.e. exactly those worlds that were present before the append operations started.
Hence each world in $\mathcal D_{i-1}(j)$ contributes exactly one extended world containing $(u_i,\alpha_i)$, and none of the newly appended worlds is reused again during the same update.

Therefore, after iteration $i$, the content of $dp[j]$ is exactly:
\begin{itemize}
\item all worlds in $\mathcal D_{i-1}(j)$, corressponding to the case $(u_i,\alpha_i)\notin W$;
\item together with all worlds of the form $W'\cup\{(u_i,\alpha_i)\}$ for
$W'\in\mathcal D_{i-1}(j-t(u_i))$, corresponding to the case $(u_i,\alpha_i)\in W$.
\end{itemize}
By the disjoint case analysis above, this is precisely $\mathcal D_i(j)$.
Thus
\[
dp[j]=\mathcal D_i(j).
\]

Since both cases have been established, \eqref{eq:sumdp-invariant} holds after iteration $i$.
This completes the proof by induction.

Hence, after the outer loop terminates, for every $j\in\{0,\dots,J\}$,
\[
dp[j]
=
\{\,W\subseteq U\mid \sum_{(u,\alpha)\in W}t(u)=j\,\}.
\]

\smallskip\noindent\textbf{Selecting satisfying worlds.}
The algorithm forms $\mathcal W$ as a union of those $dp[j]$ whose index $j$ satisfies the comparison with $C$:
\begin{itemize}
\item if $\op$ is $=$, it takes $dp[C]$;
\item if $\op$ is $>$, it takes $\bigcup_{j=C+1}^{J} dp[j]$;
\item if $\op$ is $<$, it takes $\bigcup_{j=0}^{C-1} dp[j]$;
\item if $\op$ is $\ge$, it takes $\bigcup_{j=C}^{J} dp[j]$;
\item if $\op$ is $\le$, it takes $\bigcup_{j=0}^{C} dp[j]$.
\end{itemize}
By the characterization of $dp[j]$ proved above, each such union is exactly the set of worlds
$W\subseteq U$ whose sum satisfies
\[
\sum_{(u,\alpha)\in W} t(u)\op C.
\]
The only possible extra world is $\emptyset$, which belongs to $dp[0]$ and may therefore appear whenever the selected range contains $0$.
The algorithm thomits it by setting
\[
\mathcal W\gets \mathcal W\setminus \{\emptyset\}.
\]
It follows that the final output is exactly
\[
\{\,W\subseteq U\mid W\neq\emptyset,\sum_{(u,\alpha)\in W} t(u)\op C\,\}.
\]

This establishes the correctness.
\end{appendixproof}
\begin{algorithm}
  \caption{Enumeration of valid worlds for $(t:\mathrm{COUNT}) \op C$ at a fixed group key $\vec v$}
  \label{alg:count_enum}

  \Input{a finite set of group-member occurrences $U^{\hat I}_{q',\vec v}$, a term $t$ with max-index $\leq k$, a constant $C\in\NN$, and an operator $\op\in \{=,\neq,>,<,\ge,\le\} $}
  \Output{a list $\mathcal W$ of worlds $\emptyset\neq W\subseteq U^{\hat I}_{q',\vec v}$ such that $|W| \op C$}

  \BlankLine
  Represent $U^{\hat I}_{q',\vec v}$ as a finite list of distinct occurrences $\big((u_i,\alpha_i)\big)_{1\le i\le N}$

  \BlankLine
  $\mathcal W \gets \emptyset$\;
  
  \SetKwFunction{Combinations}{Combinations}
  \Fn{\Combinations{$i, x, W$}}{
    \If{$x = 0$}{
      append $W$ to $\mathcal W$\;
      \Return\;
    }
    \If{$i > N$}{
      \Return\;
    }
    \If{$N-i+1 < x$}{
      \Return\;
    }

    \tcp{Case 1: do not select occurrence $i$}
    \Combinations{$i+1, x, W$}\;

    \tcp{Case 2: select occurrence $i$}
    \Combinations{$i+1, x-1, W \cup \{(u_i,\alpha_i)\}$}\;
  }

  \SetKwFunction{AddExact}{AddExact}
  \Fn{\AddExact{$x$}}{
    \If{$x\le 0$ \textbf{or} $x>N$}{\Return}
    \Combinations{$1, x, \emptyset$}\;
     }

  \BlankLine
  \Switch{$\op$}{
    \Case{$=$}{
      \AddExact{$C$}\;
    }
    \Case{$>$}{
      \For{$x\gets C+1$ \KwTo $N$}{\AddExact{$x$}}
    }
    \Case{$\ge$}{
      \For{$x\gets C$ \KwTo $N$}{\AddExact{$x$}}
    }
    \Case{$<$}{
      \For{$x\gets 1$ \KwTo $\min(C-1,N)$}{\AddExact{$x$}}
    }
    \Case{$\le$}{
      \For{$x\gets 1$ \KwTo $\min(C,N)$}{\AddExact{$x$}}
    }
    \Case{$\neq$}{
      \For{$x\gets 1$ \KwTo $N$}{
        \If{$x\neq C$}{\AddExact{$x$}}
      }
    }
  }

  \Return $\mathcal W$\;
\end{algorithm}
\begin{theoremrep}[Correctness of Algorithm~\ref{alg:count_enum}]\label{thm:count-enum-correct}
Fix a $\mathbb K$-instance $\hat I$ and a group key $\vec v$.
Let $U\defeq U^{\hat I}_{q',\vec v}$ and represent it as a finite list of distinct occurrences
$((u_i,\alpha_i))_{1\le i\le N}$.
Let $t$ be a term of max-index $\le k$.
Fix $C\in\NN$ and $\op\in\{=,\neq,>,<,\ge,\le\}$.
Let $\mathcal W$ be the output of Algorithm~\ref{alg:count_enum} on input $(U,t,C,\op)$.
Then
\[
\mathcal W
=
\{\,W\subseteq U\mid W\neq\emptyset,\ |W|\op C\,\}.
\]
\end{theoremrep}

\begin{appendixproof}
The term $t$ is part of the input signature for uniformity with the other algorithm which was for SUM aggregate, but the computation for \textsc{COUNT} depends only on the occurrences in $U$.

We represent $U$ as the finite list of distinct occurrences
\[
((u_i,\alpha_i))_{1\le i\le N}.
\]
\smallskip\noindent\textbf{Correctness of \textsc{Combinations}.}
For integers $i$ and $x$, and for a $W\subseteq \{(u_1,\alpha_1),\dots,(u_{i-1},\alpha_{i-1})\}$, consider a call
\[
\textsc{Combinations}(i,x,W).
\]
We claim that this call appends to $\mathcal W$ exactly the sets
\[
W'\subseteq U
\]
such that:
\begin{itemize}
\item $W\subseteq W'$;
\item every element of $W'\setminus W$ belongs to $\{(u_i,\alpha_i),\dots,(u_N,\alpha_N)\}$;
\item $|W'\setminus W|=x$.
\end{itemize}
Equivalently, it appends exactly the worlds of the form
\[
W\cup S
\quad\text{with}\quad
S\subseteq \{(u_i,\alpha_i),\dots,(u_N,\alpha_N)\}
\text{ and } |S|=x.
\]

We prove this by induction on the pair $(N-i+1,x)$.

\smallskip\noindent\emph{Base cases.}
If $x=0$, it appends $W$ and returns. This is correct because the only way to extend $W$ by choosing exactly $0$ further occurrences is to keep $W$.

If $i>N$, there are no remaining occurrences to choose from. Hence no extension of $W$ by a +venumber of further occurrences exists, so returning without appending anything is correct.

If $N-i+1<x$, then fewer than $x$ occurrences remain available. Thus it is impossible to choose exactly $x$ more occurrences, so returning without appending is again correct.

\smallskip\noindent\emph{Inductive step.}
Assume now that $x>0$, $i\le N$, and $N-i+1\ge x$.
Any extension of $W$ by exactly $x$ occurrences chosen from
\[
\{(u_i,\alpha_i),\dots,(u_N,\alpha_N)\}
\]
falls into one of two disjoint cases:
\begin{itemize}
\item it does \emph{not} contain $(u_i,\alpha_i)$;
\item it \emph{does} contain $(u_i,\alpha_i)$.
\end{itemize}

In the first case, the required worlds are exactly those produced by the recursive call
\[
\textsc{Combinations}(i+1,x,W),
\]
since one must still choose all $x$ required occurrences from the suffix starting at $i+1$.

In the second case, the desired worlds are exactly those produced by
\[
\textsc{Combinations}(i+1,x-1,W\cup\{(u_i,\alpha_i)\}),
\]
since after selecting $(u_i,\alpha_i)$, one must select exactly $x-1$ further occurrences from the suffix starting at $i+1$.

By the I.H, the first recursive call appends exactly the words of the first kind, and the second appends exactly the words of the second kind. Since the two cases are disjoint and exhaustive, \textsc{Combinations} appends exactly all desired words, each exactly once.

This proves the claim.

\smallskip\noindent\textbf{Correctness of \textsc{AddExact}.}
Consider a call \textsc{AddExact}$(x)$.

If $x\le 0$, the procedure returns immediately. This is correct because the output requires $W\neq\emptyset$, so no word of size $0$ should be output, and no word has negative cardinality.

If $x>N$, the procedure also returns immediately. This is correct because no subset of $U$ can have cardinality greater than $N$.

If $1\le x\le N$, the procedure calls
\[
\textsc{Combinations}(1,x,\emptyset).
\]
By the claim proved above, this appends exactly all sets
\[
W\subseteq U
\quad\text{such that}\quad
|W|=x.
\]
Hence \textsc{AddExact}$(x)$ appends exactly the non-empty words of cardinality $x$.

\smallskip\noindent\textbf{Selecting satisfying worlds.}
We now inspect the final case distinction on $\op$.

\begin{itemize}
\item If $\op$ is $=$, the algorithm executes \textsc{AddExact}$(C)$, so by the correctness of \textsc{AddExact} it outputs exactly the non-empty worlds $W\subseteq U$ such that $|W|=C$.

\item If $\op$ is $>$, the algorithm executes \textsc{AddExact}$(x)$ for every
\[
x\in\{C+1,\dots,N\}.
\]
Therefore it returns exactly the non-empty worlds whose cardinality is strictly greater than $C$.

\item If $\op$ is $\ge$, the algorithm executes \textsc{AddExact}$(x)$ for every
\[
x\in\{C,\dots,N\}.
\]
Since \textsc{AddExact} ignores values $x\le 0$, this outputs exactly the non-empty worlds whose cardinality is at least $C$.

\item If $\op$ is $<$, the algorithm executes \textsc{AddExact}$(x)$ for every
\[
x\in\{1,\dots,\min(C-1,N)\}.
\]
Thus it returns exactly the non-empty worlds whose cardinality is strictly less than $C$.

\item If $\op$ is $\le$, the algorithm executes \textsc{AddExact}$(x)$ for every
\[
x\in\{1,\dots,\min(C,N)\}.
\]
Thus it retunrs exactly the non-empty worlds whose cardinality is at most $C$.

\item If $\op$ is $\neq$, the algorithm executes \textsc{AddExact}$(x)$ for every
\[
x\in\{1,\dots,N\}\setminus\{C\}.
\]
Hence it returns exactly the non-empty worlds whose cardinality is different from $C$.
\end{itemize}

In every case, the resulting  $\mathcal W$ is exactly
\[
\{\,W\subseteq U\mid W\neq\emptyset,\ |W|\op C\,\}.
\]

This establishes correctness.
\end{appendixproof}

Some important cases of aggregate comparisons can be evaluated in polynomial time:
\begin{proposition}
  The provenance of HAVING queries of the following forms can be computed
  in polynomial time in data complexity:
  \begin{itemize}
    \item \sql{MIN(a)}${}\op c$,
    \sql{MAX(a)}${}\op c$,
\fakesql{PICKFIRST}\texttt{(a)}${}\op c$
      with
      $\op\in\{<,\leq,=,\geq,>,\neq\}$ for any arbitrary m-semiring;
    \item \sql{COUNT(*) >= k} and \sql{COUNT(*) > k} for $k$ a fixed constant, in absorptive
      m-semirings;
\item \sql{SUM(a) >= }$c$ and \sql{SUM(a) > }$c$ with $c\geq 0$ when for every value $a$
  of the attribute \texttt{a}, $\frac ca\leq k$ where $k\geq 0$ is a fixed constant, in absorptive
      m-semirings;
    \item \sql{COUNT(*) = k}, \sql{COUNT(*) <= k}, and \sql{COUNT(*) < k} for $k$ a fixed
      constant, in arbitrary m-semirings.
  \end{itemize}
\end{proposition}
\begin{proof}
Fix an arbitrary $\mathbb K$-instance $\hat I$, a group key $\vec v$, and represnet
\[
U^{\hat I}_{q',\vec v}
=
\{(u_1,\alpha_1),\dots,(u_N,\alpha_N)\}.
\]
For each case below, we show that the provenance for $\vec v$ can be computed in time polynomial in $N$. Since the queries are fixed, this implying polynomial time in data complexity.
We recall that the provenance of the HAVING condition( eqv. selection predicates with aggregate comparisons ) at group key $\vec v$ is obtained by enumerating the valid non-empty worlds
and then forming the corresponding semiring sum of the $\mathbb K$-annotations associated with these worlds using the possible world semantics. Thus it suffices to show that, in each of the stated cases, the relevant list of worlds can be computed in polynomial time.

\smallskip\noindent\textbf{Case 1: \sql{MIN(a)}$\op c$, \sql{MAX(a)}${}\op c$, and \fakesql{PICKFIRST}\texttt{(a)}$\op c$.}
For these aggregates, validity of a world is determined directly from the occurrences in $U^{\hat I}_{q',\vec v}$:
\begin{itemize}
  \item for \sql{MIN(a)}${}\op c$, one checks whether the selected occurrences contain an admissible minimum value and exclude all values that would violate the comparison;
  \item \sql{MAX(a)}${}\op c$ is a dual to \sql{MIN(a)}${}\op c$;
  \item for \fakesql{PICKFIRST}\texttt{(a)}$\op c$, one checks which occurrence is the first and if its value satisfies the comparison.\end{itemize}
The provenance can be computed by a direct scan over the occurrences in the group, using only polynomially many operations in $N$. Therefore these cases are in polynomial time for arbitrary m-semirings.

\smallskip\noindent\textbf{Case 2: \sql{COUNT(*) >= k} and \sql{COUNT(*) > k} for fixed $k$, in absorptive m-semirings.}
By correctness of Algorithm~\ref{alg:count_enum}, the valid worlds are exactly the non-empty subsets
$W\subseteq U^{\hat I}_{q',\vec v}$ such that $|W|\ge k\text{ or }|W|>k$, respectively.
  If such a $W$ satisfies the condition, then every superset of $W$ also satisfies it; equivalently these $W$ form an upward-closed family of sets. The m-semiring $\mathbb K$ being absorptive, the provenance of an upward-closed family of sets is equal to the provenance of its family of minimal sets. Here, The minimal satisfying worlds are:
\begin{itemize}
\item exactly the worlds of size $k$ for \sql{COUNT(*) >= k};
\item exactly the worlds of size $k+1$ for \sql{COUNT(*) > k}.
\end{itemize}
Since $k$ is fixed, the number of such worlds is $\binom{N}{k}\text{ or }\binom{N}{k+1}$,
which is polynomial in $N$. Therefore the provenance can be computedi n polynomial time.

\smallskip\noindent\textbf{Case 3: \sql{SUM(a) >= c} and \sql{SUM(a) > c} with $c\ge 0$, in absorptive m-semirings.}
By correctness of Algorithm~\ref{alg:sum_dp}, the valid worlds are exactly the non-empty subsets
$W\subseteq U^{\hat I}_{q',\vec v}$ such that $\sum_{(u,\alpha)\in W} a(u)\ge c\text{ or }\sum_{(u,\alpha)\in W} a(u)>c$ .
Since all values are nonnegative, these families are upward-closed under inclusion. Hence, in an absorptive m-semiring, it suffices to consider the minimal satisfying worlds.
Let $W$ be a minimal satisfying world. If Each selected occurrence contributes at least $c/k$, then the size of $W$ is at most $k$: indeed, if $|W|>k$, then the sum of the values in $W$ is already forced by more than $k$ nonnegative contributions each at least $c/k$, contradicting minimality of $W$. Therefore every minimal satisfying $W$ has cardinality at most $k$.

Therefore the number of minimal satisfying worlds is bounded by $\sum_{i=1}^{k}\binom{N}{i}$ ,
which is polynomial in $N$, $k$ being fixed. Therefore the provenance can be computed in polynomial time.

\smallskip\noindent\textbf{Case 4: \sql{COUNT(*) = k}, \sql{COUNT(*) <= k}, and \sql{COUNT(*) < k} for fixed $k$, in arbitrary m-semirings.}
By correctness of Algorithm~\ref{alg:count_enum}, the valid worlds are exactly:
\begin{itemize}
\item the worlds of size exactly $k$ for \sql{COUNT(*) = k};
\item the worlds of sizes $1,\dots,k$ for \sql{COUNT(*) <= k};
\item the worlds of sizes $1,\dots,k-1$ for \sql{COUNT(*) < k}.
\end{itemize}
Since $k$ is fixed, the number of such worlds is bounded by
  $\binom{N}{k},\, \sum_{i=1}^{k}\binom{N}{i},\text{ and } \sum_{i=1}^{k-1}\binom{N}{i}$, respectively. Each of these bounds is polynomial in $N$. Therefore the provenance can be computed in polynomial time in arbitrary m-semirings.

\smallskip\noindent
WIth these 4 cases, provenance can be computed in polynomial time for all the HAVING queries describe in the proposition.
\end{proof}
