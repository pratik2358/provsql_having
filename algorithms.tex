\section{Algorithms}
\label{sec:algorithms}

\pierre{2 pages}
In this section we describe specialized procedures for evaluating the possible-world
contributions of aggregate comparison-atoms inside $\predsem{\hat I}{q}{\psi}$.


\smallskip\noindent
In the implementation, ProvSQL represents multiset semantics using \emph{tuple occurrences} carrying globally unique
provenance identifiers(UUIDs). We maintain a uniqueness-invariant that for every intermediate query result
$\angsem{\hat I}{q}$, no pair $(u,\alpha)$ occurs twice: duplicates of the same value tuple $u$ are represented
as distinct occurrences with different annotations $\alpha$. Under this invariant, the multiset
$U^{\hat I}_{q',\vec v}$ of group-member occurrences can be treated as a finite \emph{set} of occurrences, and
possible worlds are ordinary subsets $W\subseteq U^{\hat I}_{q',\vec v}$. This does not change the semantics:
multiset properties are preserved because distinct occurrences may still yield equal values $t(u)$, so the induced
aggregate input $\vals_t(W)=\mset{t(u)\mid (u,\alpha)\in W}$ remains a multiset of values.

\smallskip\noindent
All algorithms are applied \emph{per group key} $\vec v$ and operate on the set of group-member
occurrences $U^{\hat I}_{q',\vec v}$ of the \emph{pre-aggregation} subquery $q'$. Accordingly, any term $t$
handled by these procedures is evaluated on tuples produced by $q'$, and its max-index is bounded by the
arity of $q'$ (denoted by $k$ in the algorithms for readability).

\smallskip\noindent
Algorithm~\ref{alg:sum_dp} is a dynamic-programming procedure for bounded-range $\mathrm{SUM}(t)*C$ atoms. It enumerates those worlds
$W\subseteq U^{\hat I}_{q',\vec v}$ whose sum $\sum_{(u,\alpha)\in W} t(u)$(i.e. $\SUM(t)$) satisfies the comparison with the
constant $C$, i.e., precisely the worlds contributing nonzero weight to the corresponding
$\chi_{\op}$-ensured possible-world sum in the semantics of aggregate comparison atoms.
By taking $t=1$, this algorithm works also
bounded-range $\mathrm{COUNT}(t)\op C$ atoms; but for $\mathrm{COUNT}$
aggregates we additionally provide a dedicated routine exploiting the 
unit-weight sum structure in a purey combinatorial fashion. Similarly, for
$\mathrm{MIN}$/$\mathrm{MAX}$ aggregates we use ordering-based strategies that avoid full world enumeration.
As a fallback for aggregate functions or predicates not covered by the specialized cases, we revert to
brute-force enumeration of non-empty worlds $W\subseteq U^{\hat I}_{q',\vec v}$. 

\smallskip\noindent
These algorithms can be used to evaluate provenance for the other type of aggregate comparison atoms where there is a comparison between two different aggregates by evaluating one of the aggrgeate into a constant value first. Correctness of these algorithms are shown in the appendix.(\rjok{to do})

\begin{algorithm}
  \caption{DP algorithm for $\mathrm{SUM}(t)\ \op\ C$ at a fixed group key $\vec v$}
  \label{alg:sum_dp}

  \Input{a finite set of group-member occurrences $U^{\hat I}_{q',\vec v}$, a term $t$ of max-index $\le k$ such that $t(u)\in\NN \; \forall (u,\alpha)\in U^{\hat I}_{q',\vec v}$ , a constant $C\in\NN$, and an operator $\op\in\{=,>,<,\ge,\le\}$}
  \Output{a list $\mathcal W$ of worlds $W\subseteq U^{\hat I}_{q',\vec v}(W\neq\emptyset)$ such that $\sum_{(u,\alpha)\in W} t(u)\ \op\ C$}

  \BlankLine
  Represent $U^{\hat I}_{q',\vec v}$ as a finite list of distinct occurrences $\big((u_i,\alpha_i)\big)_{1\le i\le N}$\;
  $T \gets \sum_{i=1}^{N} t(u_i)$
  \If{$\op\in\{>,\ge\}$}{
    $J \gets T$\;
  }
  \ElseIf{$\op=<$}{
    $J \gets \min(C-1,\ T)$\;
  }
  \Else{
    $J \gets \min(C,\ T)$  }

  \BlankLine
  Initialize $dp[0] \leftarrow \{\emptyset\}$\;
  \For{$j \gets 1$ \KwTo $J$}{
    $dp[j] \leftarrow \emptyset$\;
  }

  \BlankLine
  $\mathrm{pref\_sum} \gets 0$\;

  \For{$i \gets 1$ \KwTo $N$}{
    $w \gets t(u_i)$\;
    $\mathrm{pref\_sum} \gets \mathrm{pref\_sum} + w$\;
    $j_{\max} \gets \min(J,\ \mathrm{pref\_sum})$\;

    \For{$j \gets j_{\max}, j_{\max}-1, \dots, w$}{
      $p \gets j - w$\;
      \If{$dp[p] \neq \emptyset$}{
        \ForEach{$S \in dp[p]$}{
          append $S \cup \{(u_i,\alpha_i)\}$ to $dp[j]$\;
        }
      }
    }
  }

  \BlankLine
  \If{$\op = =$}{
    $\mathcal W \leftarrow dp[C]$\;
  }
  \ElseIf{$\op = >$}{
    $\mathcal W \leftarrow \bigcup_{j=C+1}^{J} dp[j]$\;
  }
  \ElseIf{$\op = <$}{
    $\mathcal W \leftarrow \bigcup_{j=0}^{C-1} dp[j]$\;
  }
  \ElseIf{$\op = \ge$}{
    $\mathcal W \leftarrow \bigcup_{j=C}^{J} dp[j]$\;
  }
  \Else{
    $\mathcal W \leftarrow \bigcup_{j=0}^{C} dp[j]$\;
  }

  \Return $\mathcal W$\;
\end{algorithm}
\begin{algorithm}
  \caption{Optimised provenance for \texttt{COUNT} aggregate comparisons: $\mathrm{Prov}(\texttt{COUNT}(\Box)\star C$)}
  \label{alg:count-prov-comparisons}

  \Input{dictionary $A\leftarrow \mathrm{supp}(\texttt{COUNT}(\Box))$}
  \Output{$\mathrm{Prov}(\texttt{COUNT}(\Box) \star C)$}

  \SetKwFunction{ProdSet}{ProdSet}

  \Fn{\ProdSet{$S$}}{
    $p \gets \mathbf{1}$\;
    \ForEach{$x \in S$}{
      $p \gets p \otimes x$\;
    }
    \Return $p$\;
  }

  $P_{\star C} \gets \mathbf{0}$\;

  \If {$\star \: =\: \ge$}{
    \ForEach{$S \subseteq A^{(K)}$ such that $|S| = C$}{
      $P_{\star C} \gets P_{\star C} \oplus \textsc{ProdSet}(S)$\;
    }
  }
  \ElseIf {$\star \: =\: \le$}{
    \ForEach{$S \subseteq A^{(K)}$ such that $|S| = n-C$}{
      $S' \gets \{\bar{x}\mid x\in S\}$\;
      $P_{\star C} \gets P_{\star C} \oplus \textsc{ProdSet}(S')$\;
    }
  }
  \ElseIf {$\star \: =\: =$}{
    $P_{\le C} \gets \mathrm{Prov}(\texttt{COUNT}(\Box))\le C$\;
    $P_{\ge C} \gets \mathrm{Prov}(\texttt{COUNT}(\Box))\ge C$\;
    $P_{\star C} \gets P_{\ge C} \otimes P_{\le C}$\;
  }

  \Return $P_{\star C}$\;
\end{algorithm}
