\begin{proposition}
  The provenance of HAVING queries of the following forms can be computed
  in polynomial time in data complexity:
  \begin{itemize}
    \item \sql{MIN(a)}${}\op c$,
    \sql{MAX(a)}${}\op c$,
\fakesql{PICKFIRST}\texttt{(a)}${}\op c$
      with
      $\op\in\{<,\leq,=,\geq,>,\neq\}$ for any arbitrary m-semiring;
    \item \sql{COUNT(*) >= k} and \sql{COUNT(*) > k} for $k$ a fixed constant, in absorptive
      m-semirings;
\item \sql{SUM(a) >= }$c$ and \sql{SUM(a) > }$c$ with $c\geq 0$ when for every value $a$
  of the attribute \texttt{a}, $\frac ca\leq k$ where $k\geq 0$ is a fixed constant, in absorptive
      m-semirings;
    \item \sql{COUNT(*) = k}, \sql{COUNT(*) <= k}, and \sql{COUNT(*) < k} for $k$ a fixed
      constant, in arbitrary m-semirings.
  \end{itemize}
\end{proposition}

\begin{algorithm}
  \tcp{\textcolor{red}{We need to redefine $x$ in this case to maintain uniformity with the paper.}}
  \caption{DP algorithm for $\mathdutchcal{u}\op C$ ($\op \in =,>,<,\ge,\le$) with bounded range and prefix-sum pruning}
  \label{alg:dp_subsets}

  \Input{a dictionary $A$ of tuples involved and $\mathbb{K}$-annotations, a constant $C \in M$, and an operator $\op\in\{=,>,<,\ge,\le\}$}
  \Output{a list $\mathcal{R}$ of subsets of $A$, s.t.\ $\forall R \in \mathcal{R}$, $\ \sum_{\mathbf{x}\in R}\mathbf{x}^{(M)}\ \op\ C$}

  \BlankLine
  $T \gets \sum_{\mathbf{x}\in A}\mathbf{x}^{(M)}$\tcp*[r]{max reachable sum}

  \If{$\op\in\{>,\ge\}$}{
    $J \gets T$\;
  }
  \ElseIf{$\op=<$}{
    $J \gets \min(C-1,\ T)$\;
  }
  \Else{
    $J \gets \min(C,\ T)$\tcp*[r]{$\op\in\{=,\le\}$}
  }

  \BlankLine
  Initialize $dp[0] \leftarrow \{\emptyset\}$\;
  \For{$j \gets 1$ \KwTo $J$}{
    $dp[j] \leftarrow \emptyset$\;
  }

  \BlankLine
  $\mathrm{pref\_sum} \gets 0$\tcp*[r]{prefix-sum pruning state}

  \For{$i \gets 1$ \KwTo $|A|$}{
    $w \gets A^{(M)}[i]$, $g \gets A^{(K)}[i]$\;
    $\mathrm{pref\_sum} \gets \mathrm{pref\_sum} + w$\;
    $j_{\max} \gets \min(J,\ \mathrm{pref\_sum})$\tcp*[r]{max reachable sum at step $i$}

    \For{$j \gets j_{\max}, j_{\max}-1, \dots, w$}{
      $p \gets j - w$\;
      \If{$dp[p] \neq \emptyset$}{
        \ForEach{$S \in dp[p]$}{
          append $S \cup \{g\}$ to $dp[j]$\;
        }
      }
    }
  }

  \BlankLine
  \If{$\op = =$}{
    $\mathcal{R} \leftarrow dp[C]$\;
  }
  \ElseIf{$\op = >$}{
    $\mathcal{R} \leftarrow \bigcup_{j=C+1}^{J} dp[j]$\;
  }
  \ElseIf{$\op = <$}{
    $\mathcal{R} \leftarrow \bigcup_{j=1}^{C-1} dp[j]$\;
  }
  \ElseIf{$\op = \ge$}{
    $\mathcal{R} \leftarrow \bigcup_{j=C}^{J} dp[j]$\;
  }
  \Else{
    $\mathcal{R} \leftarrow \bigcup_{j=1}^{C} dp[j]$\;
  }

  \Return $\mathcal{R}$\;
\end{algorithm}

\begin{algorithm}
  \caption{Probability computation for $[\mathdutchcal{u}\oslash C]_{\op}$}
  \label{alg:probability}

  \Input{dictionary $A\leftarrow\mathrm{supp}(\mathdutchcal{u})$, constant $C\in M$, operator $\op$, and probabilities $p(g)$ for all $g\in\mathcal{G}$}
  \Output{$\Pr\big([\mathdutchcal{u}\oslash C]_{\op}\big)$}

  $\mathcal{R} \gets \textsc{DP-Subsets}(A,C,\op)$\tcp*[r]{Algorithm above}
  $P \gets 0$\;
  \ForEach{$S \in \mathcal{R}$}{
    $\pi \gets 1$\;
    \ForEach{$g \in \mathcal{G}$}{
      \If{$g \in S$}{
        $\pi \gets \pi \cdot p(g)$\;
      }
      \Else{
        $\pi \gets \pi \cdot (1 - p(g))$\;
      }
    }
    $P \gets P + \pi$\;
  }
  \Return $P$\;
\end{algorithm}

\begin{algorithm}
  \caption{Optimised provenance for \texttt{COUNT} aggregate comparisons: $\mathrm{Prov}(\texttt{COUNT}(\Box)\star C$)}
  \label{alg:count-prov-comparisons}

  \Input{dictionary $A\leftarrow \mathrm{supp}(\texttt{COUNT}(\Box))$}
  \Output{$\mathrm{Prov}(\texttt{COUNT}(\Box) \star C)$}

  \SetKwFunction{ProdSet}{ProdSet}

  \Fn{\ProdSet{$S$}}{
    $p \gets \mathbf{1}$\;
    \ForEach{$x \in S$}{
      $p \gets p \otimes x$\;
    }
    \Return $p$\;
  }

  $P_{\star C} \gets \mathbf{0}$\;

  \If {$\star \: =\: \ge$}{
    \ForEach{$S \subseteq A^{(K)}$ such that $|S| = C$}{
      $P_{\star C} \gets P_{\star C} \oplus \textsc{ProdSet}(S)$\;
    }
  }
  \ElseIf {$\star \: =\: \le$}{
    \ForEach{$S \subseteq A^{(K)}$ such that $|S| = n-C$}{
      $S' \gets \{\bar{x}\mid x\in S\}$\;
      $P_{\star C} \gets P_{\star C} \oplus \textsc{ProdSet}(S')$\;
    }
  }
  \ElseIf {$\star \: =\: =$}{
    $P_{\le C} \gets \mathrm{Prov}(\texttt{COUNT}(\Box))\le C$\;
    $P_{\ge C} \gets \mathrm{Prov}(\texttt{COUNT}(\Box))\ge C$\;
    $P_{\star C} \gets P_{\ge C} \otimes P_{\le C}$\;
  }

  \Return $P_{\star C}$\;
\end{algorithm}
